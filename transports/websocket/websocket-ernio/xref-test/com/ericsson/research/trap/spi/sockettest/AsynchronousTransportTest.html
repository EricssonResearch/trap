123     private void testContext(TrapEndpoint ep) throws TrapException
124     {
125         TrapTransport t = ep.getTransport("websocket");
126         Collection<String> keys = t.getAuthenticationKeys();
127         
128         Assert.assertTrue(keys.containsAll(Arrays.asList(Transport, Protocol, LocalIP, LocalPort, RemoteIP, RemotePort, State, Format)));
129         
130         Map<String, Map<String, Object>> cx = ep.getTransportAuthenticationContexts();
131         
132         Assert.assertTrue(cx.containsKey("websocket"));
133         
134         Map<String, Object> props = cx.get("websocket");
135         
136         Set<String> returnedKeys = props.keySet();
137         
138         for (String key : keys)
139             Assert.assertTrue("Key: " + key, returnedKeys.contains(key));        
140     }

202     public void performMessageTests(final int messages) throws Exception
203     {
204         
205         final byte[] bytes = "Helloes".getBytes();
206         
207         this.receiving = new AtomicInteger(4);
208         this.receipts = new ConcurrentLinkedQueue<byte[]>();
209         this.receivingCount = new AtomicInteger(0);
210         this.processed = new AtomicInteger(0);
211         this.messages = messages;
212         
213         for (int k = 0; k < this.receiving.get(); k++)
214         {
215             ThreadPool.executeFixed(new Runnable() {
216                 
217                 public void run()
218                 {
219                     for (int i = 0; i < (messages / AsynchronousTransportTest.this.receiving.get()); i++)
220                         try
221                         {
222                             byte[] b = AsynchronousTransportTest.this.receive();
223                             if (b == null)
224                                 continue;
225                             Assert.assertArrayEquals(b, bytes);
226                             AsynchronousTransportTest.this.processed.incrementAndGet();
227                         }
228                         catch (Exception e)
229                         {
230                             e.printStackTrace();
231                         }
232                     AsynchronousTransportTest.this.receiving.decrementAndGet();
233                 }
234             });
235         }
236         
237         ThreadPool.executeCached(new Runnable() {
238             
239             public void run()
240             {
241                 
242                 try
243                 {
244                     AsynchronousTransportTest.this.s.send(bytes);
245                     Thread.sleep(200);
246                 }
247                 catch (TrapException e1)
248                 {
249                     e1.printStackTrace();
250                 }
251                 catch (InterruptedException e)
252                 {
253                     e.printStackTrace();
254                 }
255                 for (int i = 1; i < messages; i++)
256                 {
257                     try
258                     {
259                         AsynchronousTransportTest.this.s.send(bytes);
260                     }
261                     catch (Throwable e)
262                     {
263                         e.printStackTrace();
264                     }
265                 }
266                 
267             }
268             
269         });
270         
271         while (this.receiving.get() != 0)
272             Thread.sleep(100);
273         
274     }

276     @Test
277     public void testLiveness() throws Exception
278     {
279         
280         this.s.send("Hello".getBytes());
281         
282         // Now check liveness
283         Assert.assertTrue(this.s.isAlive(100, true, false, 0).get());
284         Assert.assertTrue(this.s.isAlive(1000, false, false, 0).get());
285         
286         Thread.sleep(10);
287         // This should return false; we're basically asking if there was a message in the last 0 milliseconds, after explicitly sleeping
288         Assert.assertFalse(this.s.isAlive(0, false, false, 0).get());
289         
290         Thread.sleep(10);
291         this.c.send("Hello".getBytes());
292         Thread.sleep(25);
293         
294         // This should succeed; the server has received a message within the last 25 ms.
295         Assert.assertTrue(this.s.isAlive(45, false, false, 0).get());
296         
297         // This should fail; the client has not.
298         Assert.assertFalse(this.c.isAlive(5, false, false, 0).get());
299     }

301     protected synchronized TrapEndpoint accept() throws InterruptedException
302     {
303         try
304         {
305             while (this.incomingEP == null)
306                 this.wait();
307             
308             return this.incomingEP;
309         }
310         finally
311         {
312             this.incomingEP = null;
313         }
314     }
315     
316     private byte[] receive() throws Exception
317     {
318         byte[] b = null;
319         while (((b = this.receipts.poll()) == null) && (this.processed.get() != this.messages))
320         {
321             Thread.sleep(1);
322         }
323         return b;
324     }

326     public synchronized void incomingTrapConnection(TrapEndpoint endpoint, TrapListener listener, Object context)
327     {
328         //System.out.println("Incoming Connection");
329         this.incomingEP = endpoint;
330         endpoint.setDelegate(this, true);
331         this.notify();
332     }

334     public void trapData(byte[] data, int channel, TrapEndpoint endpoint, Object context)
335     {
336         //System.out.println(new String(data));
337         this.receivingCount.incrementAndGet();
338         this.receipts.add(data);
339         
340         if (this.receipts.size() > 10000)
341             try
342             {
343                 Thread.sleep(10);
344             }
345             catch (InterruptedException e)
346             {
347                 e.printStackTrace();
348             }
349     }
