52  public class StaticKeepalivePredictor implements TrapKeepalivePredictor, Runnable
53  {
54      
55      protected int                                keepaliveInterval   = TrapKeepalivePolicy.DISABLED;
56      
57      // Keepalive engine stuff
58      /**
59       * The current keepalive interval
60       */
61      protected int                                mKeepaliveInterval  = 5 * 60;
62      
63      /**
64       * Number of seconds to wait at least between keepalives
65       */
66      protected int                                minKeepalive        = 1;
67      
68      /**
69       * Number of seconds to wait at most between keepalives
70       */
71      protected int                                maxKeepalive        = 999999;
72      
73      // Automatic keepalive interval optimisation
74      
75      protected int                                lastInterval        = this.mKeepaliveInterval;
76      
77      protected int                                growthStep          = 0;
78      
79      protected int                                nextInterval        = this.mKeepaliveInterval + this.growthStep;
80      
81      /**
82       * The minimum keepalive value that the automatic keepalive algorithm is allowed to decrease the keepalive to. The
83       * auto keepalive algorithm is only active on transports that can connect (i.e. reconnect) if it fails them.
84       */
85      protected int                                minAutoKeepalive    = 1;
86      
87      /**
88       * The minimum keepalive value that the automatic keepalive algorithm is allowed to decrease the keepalive to. The
89       * auto keepalive algorithm is only active on transports that can connect (i.e. reconnect) if it fails them. In
90       * addition, if keepalivepolicy != default, the automatic algorithm is disabled, as well as when min/max auto
91       * keepalives are negative numbers.
92       */
93      protected int                                maxAutoKeepalive    = 28 * 60;
94      
95      /**
96       * Timestamp of last recorded keepalive received
97       */
98      protected long                               lastDataReceived    = 0;
99      protected long                               lastDataSent        = 0;
100     protected long                               lastSentKeepalive   = 0;
101     
102     protected Future                             keepaliveTask       = null;
103     protected long                               keepaliveTaskTime   = 0;
104     protected long                               keepaliveExpiryMsec = 5000;
105     
106     /**
107      * Byte array containing the most recently sent keepalive message from this predictor.
108      */

137     public synchronized void setKeepaliveInterval(int interval)
138     {
139         this.keepaliveInterval = interval;
140         
141         if (interval == TrapKeepalivePolicy.DEFAULT)
142             this.nextInterval = this.mKeepaliveInterval;
143         else if (interval == TrapKeepalivePolicy.DISABLED)
144             this.nextInterval = -1;
145         else
146         {
147             // Basically, ensure that the interval is within the allowed range
148             if ((interval > this.maxKeepalive) || (interval < this.minKeepalive))
149                 this.nextInterval = this.mKeepaliveInterval;
150             else
151                 this.nextInterval = interval;
152         }
153         
154         // Reschedule if we are started, to take into account the new interval.
155         if (this.started)
156             this.schedule();
157         
158     }
159     
160     public synchronized int getKeepaliveInterval()
161     {
162         return this.keepaliveInterval;
163     }
164     
165     public synchronized long getNextKeepaliveSend()
166     {
167         // This is the oldest timestamp of data received or sent. In the case of an asynchronous transfer
168         // (e.g. only receiving data), we must even it out both ways to ensure the connection is both-ways-alive.
169         long earliestTime = Math.min(this.lastDataReceived, this.lastDataSent);
170         long expected = earliestTime + (this.nextInterval * 1000);
171         long actual = System.currentTimeMillis();
172         return expected - actual;
173     }
174     
175     public void keepaliveReceived(boolean isPing, char pingType, int timer, byte[] data)
176     {
177         if (this.logger.isTraceEnabled())
178             this.logger.trace("Received new keepalive; boolean: {}, pingType: {}, timer: {}, data: {}", new Object[] { isPing, pingType, timer, HexDump.makeStringifyable(data, 0, 40) });
179         if (!isPing)
180         {
181             synchronized (this)
182             {
183                 // Check if this is a PING we have sent
184                 if (!Arrays.equals(data, this.keepaliveData))
185                 {
186                     if (this.keepaliveData == null)
187                     {
188                         this.logger.trace("No PING sent; ignoring keepalive");
189                         return;
190                     }
191                     
192                     if (data == null)
193                     {
194                         throw new NullPointerException();
195                     }
196                     return;
197                 }
198                 
199                 this.keepaliveData = null;
200                 
201                 switch (pingType)
202                 {
203                 // Keepalives disabled
204                 
205                     case '1':
206                         break; // Do nothing; we will not auto-adjust
207                         
208                     case '2':
209                         this.setKeepaliveInterval(timer); // Manual adjustment
210                         break;
211                     
212                     case '3': // Manually triggered keepalive
213                         break;
214                     
215                     default: // no-error
216                 }
217             }
218             
219             // Now reschedule ourselves. The received keepalive will already have been recorded as per dataReceived()
220             this.schedule();
221         }
222         else
223         {
224             TrapKeepaliveDelegate dObj = this.delegate.get();
225             
226             if (pingType == '2')
227                 this.setKeepaliveInterval(timer);
228             
229             if (dObj != null)
230                 dObj.shouldSendKeepalive(false, this.getPingType(), timer, data);
231         }
232     }
233     
234     public synchronized long nextKeepaliveReceivedDelta()
235     {
236         if (this.keepaliveData == null)
237             return Long.MAX_VALUE; // NEVER! We haven't sent a keepalive so we don't expect one. DUH.
238             
239         long expected = this.lastSentKeepalive;
240         long actual = System.currentTimeMillis();
241         return expected - actual;
242         
243     }

261     public synchronized void start()
262     {
263         if (this.started)
264         {
265             return;
266         }
267         
268         if (this.keepaliveInterval == TrapKeepalivePolicy.DISABLED)
269             return; // Don't start
270             
271         if (this.logger.isTraceEnabled())
272             this.logger.trace("Starting keepalive task for transport " + this.delegate.get() + "with predictor " + this + " and policy " + this.keepaliveInterval);
273         
274         this.keepaliveData = null;
275         this.lastSentKeepalive = Long.MAX_VALUE;
276         this.keepaliveTaskTime = 0;
277         this.lastDataReceived = this.lastDataSent = System.currentTimeMillis();
278         
279         this.started = true;
280         
281         this.schedule();
282         
283     }
284     
285     public synchronized void stop()
286     {
287         
288         if (!this.started)
289             return;
290         
291         if (this.keepaliveTask != null)
292         {
293             this.keepaliveTask.cancel();
294             this.keepaliveTask = null;
295         }
296         
297         this.started = false;
298     }
299     
300     protected void schedule()
301     {
302         
303         synchronized (this)
304         {
305             if (this.getKeepaliveInterval() == TrapKeepalivePolicy.DISABLED)
306             {
307                 this.logger.trace("Keepalive timer is disabled...");
308                 return;
309             }
310             
311             if (!this.started)
312             {
313                 this.logger.trace("Keepalive not started");
314                 return;
315             }
316             
317             // Next send should auto-disable if there is an outstanding ping/pong waiting
318             long nextSend = (this.keepaliveData == null ? this.getNextKeepaliveSend() : Long.MAX_VALUE);
319             long nextReceive = this.nextKeepaliveReceivedDelta() + (this.keepaliveData == null ? 0 : this.keepaliveExpiryMsec);
320             long msec = Math.min(nextSend, nextReceive);
321             
322             // Stop the immediate run loop. This stack overflowed anyway...
323             if (msec <= 500)
324             {
325                 msec = 501;
326             }
327             
328             long scheduledTime = msec + System.currentTimeMillis();
329             
330             // no-op: we want to schedule for longer time than the current expiry (expiry will re-schedule)
331             // cancel: we want to schedule for shorter time than the current expiry
332             if (this.keepaliveTask != null)
333             {
334                 
335                 // Already happened
336                 if (this.keepaliveTaskTime >= System.currentTimeMillis())
337                 {
338                     
339                     // Ensure we don't schedule if a task is going to happen closer, but in the future
340                     // Allow a small absolute time shift to prevent tasks from being pre-empted too often.
341                     if ((this.keepaliveTaskTime <= (scheduledTime + 250)) && (this.keepaliveTaskTime > System.currentTimeMillis()))
342                     {
343                         if (this.logger.isTraceEnabled())
344                             this.logger.trace(this + " (cached) Next keepalive due at " + new Date(this.keepaliveTaskTime));
345                         return;
346                     }
347                     this.logger.trace("Cancelling existing task...");
348                     
349                     this.keepaliveTask.cancel();
350                 }
351             }
352             
353             this.keepaliveTaskTime = scheduledTime;
354             this.keepaliveTask = ThreadPool.executeAt(new WeakRunnable(this), scheduledTime);
355             
356             if (this.logger.isTraceEnabled())
357                 this.logger.trace(this + " Next keepalive wakeup at " + new Date(this.keepaliveTaskTime));
358         }
359     }
360     
361     public synchronized void run()
362     {
363         TrapKeepaliveDelegate delegate = this.delegate.get();
364         this.logger.trace("Keepalive task timer expired. Now checking keepalives for delegate {} and policy {}.", new Object[] { delegate, this.getKeepaliveInterval() });
365         
366         if (delegate == null)
367         {
368             this.stop();
369             return; // Delegate garbage collected; nothing to keep notifying about
370         }
371         
372         // Check if we have been disabled...
373         if (this.getKeepaliveInterval() == TrapKeepalivePolicy.DISABLED)
374         {
375             this.stop();
376             return;
377         }
378         
379         // Now check for timeout
380         // This is for RECEIVING.
381         long msec = this.nextKeepaliveReceivedDelta();
382         this.logger.trace("Receive MSEC calculation: {}", Long.valueOf(msec));
383         
384         if ((msec <= 500) && (-msec > this.keepaliveExpiryMsec))
385         {
386             this.logger.trace("Notifying delegate that predicted keepalive has expired.");
387             delegate.predictedKeepaliveExpired(this, -msec);
388             this.stop();
389             return;
390         }
391         
392         // Is it time to send a keepalive?
393         
394         msec = this.getNextKeepaliveSend();
395         this.logger.trace("Send MSEC calculation: {}", Long.valueOf(msec));
396         if (msec <= 500)
397         {
398             
399             if (this.keepaliveData != null)
400             {
401                 // OOps?
402                 this.logger.trace("EXPERIMENTAL: keepalive data != null when expired timer... Dropping sending a keepalive.");
403             }
404             else
405             {
406                 this.keepaliveData = StringUtil.toUtfBytes(UUID.randomUUID());
407                 this.lastSentKeepalive = System.currentTimeMillis();
408                 this.lastInterval = this.nextInterval;
409                 delegate.shouldSendKeepalive(true, this.getPingType(), this.nextInterval, this.keepaliveData);
410                 
411             }
412         }
413         
414         // reschedule ourselves for a default time
415         this.schedule();
416     }
417     
418     private char getPingType()
419     {
420         char type = '1';
421         
422         if (this.getKeepaliveInterval() > 0)
423             type = '2';
424         
425         return type;
426     }

429     {
430         this.logger.trace("Keepalive predictor {} was garbage collected and will not predict anything else...", this);
431     }

449 class WeakRunnable implements Runnable
450 {
451     
452     private final WeakReference<Runnable> runnable;
453     
454     public WeakRunnable(Runnable r)
455     {
456         this.runnable = new WeakReference<Runnable>(r);
457     }
458     
459     public void run()
460     {
461         Runnable r = this.runnable.get();
462         
463         if (r != null)
464             r.run();
465     }
