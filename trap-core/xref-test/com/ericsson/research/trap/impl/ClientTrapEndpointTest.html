220     @Test
221     public void testEnableTransport() throws Exception
222     {
223         ClientTrapEndpoint cte = (ClientTrapEndpoint) TrapFactory.createClient("sockets://traphost.com:443", false);
224         TrapTransport tt = new TestTransport1("foo", "foo", true, true);
225         
226         if (tt.isEnabled())
227             tt.disable();
228         
229         Assert.assertFalse(tt.isEnabled());
230         
231         cte.addTransport(tt);
232         
233         Assert.assertTrue(cte.getTransports().contains(tt));
234         
235         cte.enableTransport("foo");
236         Assert.assertTrue(tt.isEnabled());
237         
238         // Ensure that re-enabling doesn't change anything
239         cte.enableTransport("foo");
240         Assert.assertTrue(tt.isEnabled());
241         
242         
243     }
