48  import java.net.InetSocketAddress;
49  import java.util.concurrent.Executors;
50  
51  import org.jboss.netty.bootstrap.ServerBootstrap;
52  import org.jboss.netty.channel.Channel;
53  import org.jboss.netty.channel.ChannelFuture;
54  import org.jboss.netty.channel.ChannelPipeline;
55  import org.jboss.netty.channel.ChannelPipelineFactory;
56  import org.jboss.netty.channel.DefaultChannelPipeline;
57  import org.jboss.netty.channel.socket.nio.NioServerSocketChannelFactory;
58  import org.jboss.netty.handler.codec.http.HttpChunkAggregator;
59  import org.jboss.netty.handler.codec.http.HttpRequestDecoder;
60  import org.jboss.netty.handler.codec.http.HttpResponseEncoder;
61  
62  import com.ericsson.research.trap.spi.transports.WebServerSocketTransport;
63  
64  /**
65   * A HTTP server which serves Web Socket requests at: http://localhost:8080/websocket Open your browser at
66   * http://localhost:8080/, then the demo page will be loaded and a Web Socket connection will be made automatically.
67   * This server illustrates support for the different web socket specification versions and will work with:
68   * <ul>
69   * <li>Safari 5+ (draft-ietf-hybi-thewebsocketprotocol-00)
70   * <li>Chrome 6-13 (draft-ietf-hybi-thewebsocketprotocol-00)
71   * <li>Chrome 14+ (draft-ietf-hybi-thewebsocketprotocol-10)
72   * <li>Chrome 16+ (RFC 6455 aka draft-ietf-hybi-thewebsocketprotocol-17)
73   * <li>Firefox 7+ (draft-ietf-hybi-thewebsocketprotocol-10)
74   * <li>Firefox 11+ (RFC 6455 aka draft-ietf-hybi-thewebsocketprotocol-17)
75   * </ul>
76   */
77  public class WebSocketServer
78  {
79      
80      public int               port;
81      
82      WebServerSocketTransport t;
83      
84      private ServerBootstrap  bootstrap;
85      
86      private Channel          bind;
87      
88      private String           host;
89      
90      public WebSocketServer(String host, int port, WebServerSocketTransport t)
91      {
92          this.host = host;
93          this.port = port;
94          this.t = t;
95      }
96      
97      public void run()
98      {
99          // Configure the server.
100         this.bootstrap = new ServerBootstrap(new NioServerSocketChannelFactory(Executors.newCachedThreadPool(), Executors.newCachedThreadPool()));
101         
102         // Set up the event pipeline factory.
103         this.bootstrap.setPipelineFactory(new ChannelPipelineFactory() {
104             
105             @Override
106             public ChannelPipeline getPipeline() throws Exception
107             {
108                 WebServerSocketHandler webServerSocketHandler = new WebServerSocketHandler();
109                 ChannelPipeline pipeline = new DefaultChannelPipeline();
110                 pipeline.addLast("decoder", new HttpRequestDecoder());
111                 pipeline.addLast("aggregator", new HttpChunkAggregator(65536));
112                 pipeline.addLast("encoder", new HttpResponseEncoder());
113                 pipeline.addLast("handler", webServerSocketHandler);
114                 
115                 WebSocketServer.this.t.notifyAccept(webServerSocketHandler);
116                 
117                 return pipeline;
118             }
119         });
120         
121         // Bind and start to accept incoming connections.
122         this.bind = this.bootstrap.bind(new InetSocketAddress(this.host, this.port));
123         this.port = ((InetSocketAddress) this.bind.getLocalAddress()).getPort();
124         
125         System.out.println("Web socket server started at port " + this.port + '.');
126         System.out.println("Open your browser and navigate to http://" + this.host + ":" + this.port + '/'); 
127     }
128     
129     public void close()
130     {
131         ChannelFuture close = this.bind.close();
132         close.syncUninterruptibly();
133     }
134 }
