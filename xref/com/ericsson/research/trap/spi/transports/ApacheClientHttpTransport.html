88      @Override
89      protected void updateConfig()
90      {
91          super.updateConfig();
92          try
93          {
94              if ((this.getState() == TrapTransportState.DISCONNECTED) || (this.getState() == TrapTransportState.CONNECTING))
95              {
96                  String url = this.getOption("url");
97                  
98                  if (url == null)
99                  {
100                     this.logger.trace("No URL specified for HTTP transport.");
101                 }
102                 else
103                 {
104                     URI uri = URI.create(url);
105                     this.connectUrl = uri;
106                 }
107             }
108             else
109                 this.logger.debug("Updating HTTP configuration while open; changes will not take effect until HTTP is reconnected");
110         }
111         catch (IllegalArgumentException e)
112         {
113             if (this.isEnabled())
114             {
115                 this.logger.debug("Invalid configuration for HTTP transport. The transport is disabled");
116                 this.connectUrl = null;
117             }
118         }
119         catch (NullPointerException e)
120         {
121             this.logger.warn("Null Pointer Exception while updating transport config. Affected config was [" + this.getConfiguration() + "]");
122         }
123         
124         String newexpirationDelay = this.getOption("expirationDelay");
125         try
126         {
127             this.expirationDelay = Long.parseLong(newexpirationDelay);
128         }
129         catch (Exception e)
130         {
131         }
132     }

140     @Override
141     public void internalSend(TrapMessage message, boolean expectMore) throws TrapTransportException
142     {
143         
144         synchronized (this.messagesToSend)
145         {
146             
147             if ((this.getState() != TrapTransportState.CONNECTED) && (this.getState() != TrapTransportState.AVAILABLE) && (this.getState() != TrapTransportState.UNAVAILABLE) && (message.getOp() != TrapMessage.Operation.CLOSE))
148                 throw new TrapTransportException(message, this.getState());
149             
150             if (message != null)
151             {
152                 if (this.logger.isTraceEnabled())
153                     this.logger.trace("[HTTP] Scheduling message {} with id {}", message.getOp(), message.getMessageId());
154                 // Don't slam messages yet
155                 this.messagesToSend.add(message);
156             }
157             
158             if (expectMore)
159                 return;
160             
161             this.flushTransport();
162             
163         }
164     }

172     @Override
173     protected void internalConnect() throws TrapException
174     {
175         
176         this.updateConfig();
177         
178         // Make a request to get a new TransportID
179         if (!this.isClientConfigured())
180         {
181             this.logger.debug("HTTP Transport not properly configured... Unless autoconfigure is enabled (and another transport succeeds) this transport will not be available.");
182             this.setState(TrapTransportState.ERROR);
183             return;
184         }
185         
186         try
187         {
188             this.running = true;
189             HttpClient httpclient = new DefaultHttpClient();
190             HttpGet httpget = this.openGet(this.connectUrl);
191             HttpResponse response = httpclient.execute(httpget);
192             HttpEntity entity = response.getEntity();
193             if (entity != null)
194             {
195                 InputStream is = null;
196                 InputStreamReader isr = null;
197                 try
198                 {
199                     is = entity.getContent();
200                     isr = new InputStreamReader(is, Charset.forName("UTF-8"));
201                     
202                     StringWriter sr = new StringWriter();
203                     int ch = 0;
204                     
205                     while ((ch = isr.read()) != -1)
206                         sr.write(ch);
207                     
208                     String rawUrl = sr.toString();
209                     
210                     String urlBase = this.connectUrl.toString();
211                     
212                     if (urlBase.endsWith("/"))
213                         urlBase += rawUrl;
214                     else
215                         urlBase += "/" + rawUrl;
216                     
217                     this.activeUrl = URI.create(urlBase);
218                     
219                     // Start a new thread for polling
220                     Thread t = new Thread(this);
221                     t.setDaemon(true);
222                     t.start();
223                     
224                     this.postclient = new DefaultHttpClient();
225                     
226                     // Change state to connected
227                     this.setState(TrapTransportState.CONNECTED);
228                 }
229                 finally
230                 {
231                     if (is != null)
232                         is.close();
233                     
234                     if (isr != null)
235                         isr.close();
236                 }
237             }
238             
239         }
240         catch (Exception e)
241         {
242             this.setState(TrapTransportState.ERROR);
243             throw new TrapException(e);
244         }
245         
246     }

248     @Override
249     protected void internalDisconnect()
250     {
251         
252         synchronized (this.messagesToSend)
253         {
254             if (this.messagesToSend.size() > 0)
255                 try
256                 {
257                     this.internalSend(null, false);
258                 }
259                 catch (TrapTransportException e)
260                 {
261                     e.printStackTrace();
262                 }
263         }
264         
265         this.running = false;
266         try
267         {
268             HttpClient httpclient = new DefaultHttpClient();
269             HttpPost post = new HttpPost(this.activeUrl);
270             HttpResponse response = httpclient.execute(post);
271             response.getEntity();
272             InputStream is = response.getEntity().getContent();
273             
274             @SuppressWarnings("unused")
275             int read = 0;
276             while ((read = is.read()) != -1)
277                 ;
278             is.close();
279         }
280         catch (Exception e)
281         {
282         }
283         this.setState(TrapTransportState.DISCONNECTED);
284     }

292     protected byte[] readBuf = new byte[4096];

294     @Override
295     public void run()
296     {
297         
298         this.logger.trace("HTTP Transport entering polling loop");
299         HttpClient httpclient = new DefaultHttpClient();
300         HttpGet httpget = null;
301         HttpResponse response = null;
302         int responseCode = -1;
303         while (this.running)
304         {
305             try
306             {
307                 
308                 httpget = this.openGet(this.activeUrl);
309                 response = httpclient.execute(httpget);
310                 
311                 responseCode = response.getStatusLine().getStatusCode();
312                 
313                 if (responseCode >= 300)
314                 {
315                     this.logger.debug("Closing due to non-200 code; Transport status is {}", this.getState());
316                     
317                     if ((this.getState() != TrapTransportState.DISCONNECTED) && (this.getState() != TrapTransportState.ERROR) && (this.getState() != TrapTransportState.DISCONNECTING))
318                         this.setState(TrapTransportState.ERROR);
319                     return;
320                 }
321                 
322                 if (responseCode == 201)
323                 {
324                     InputStream is = response.getEntity().getContent();
325                     
326                     int read = 0;
327                     while ((read = is.read(this.readBuf)) != -1)
328                     {
329                         this.receive(this.readBuf, 0, read);
330                     }
331                     
332                     is.close();
333                 }
334                 else
335                 {
336                     httpget.abort();
337                 }
338             }
339             catch (Exception e)
340             {
341                 //e.printStackTrace();
342                 this.logger.debug("Moving to state ERROR due to exception", e);
343                 this.setState(TrapTransportState.ERROR);
344             }
345         }
346         this.logger.trace("HTTP Transport exiting polling loop. Running is false.");
347     }

349     protected HttpGet openGet(URI cUrl) throws IOException
350     {
351         /*
352          * When the URL is formatted as
353          *
354          * 	http://localhost:8088
355          *
356          * Then this automatic syntax will create a non-valid url. The code
357          * below will detect if there is no path of the url; only if there is
358          * no path will it append a slash.
359          */
360         String urlStr = cUrl.toString();
361         if ((cUrl.getPath() == null) || (cUrl.getPath().trim().length() == 0))
362             urlStr = urlStr + "/";
363         
364         urlStr += "?expires=" + this.expirationDelay;
365         
366         URI uri = URI.create(urlStr);
367         return new HttpGet(uri);
368     }

380     @Override
381     public void flushTransport()
382     {
383         synchronized (this.messagesToSend)
384         {
385             try
386             {
387                 if (this.logger.isTraceEnabled())
388                     this.logger.trace("[HTTP] Flushing {} messages", this.messagesToSend.size());
389                 
390                 // Don't bother if we have no messages to send...
391                 if (this.messagesToSend.size() == 0)
392                     return;
393                 
394                 ByteArrayOutputStream bos = new ByteArrayOutputStream();
395                 
396                 for (TrapMessage m : this.messagesToSend)
397                 {
398                     bos.write(m.serialize());
399                 }
400                 
401                 byte[] body = bos.toByteArray();
402                 
403                 HttpPost post = new HttpPost(this.activeUrl);
404                 
405                 ByteArrayEntity bae = new org.apache.http.entity.ByteArrayEntity(body);
406                 post.setEntity(bae);
407                 
408                 HttpResponse response = this.postclient.execute(post);
409                 int responseCode = response.getStatusLine().getStatusCode();
410                 HttpEntity entity = response.getEntity();
411                 if (entity != null)
412                 {
413                     InputStream is = entity.getContent();
414                     
415                     @SuppressWarnings("unused")
416                     int read = 0;
417                     while ((read = is.read()) != -1)
418                         ;
419                     is.close();
420                 }
421                 
422                 if (responseCode >= 400)
423                     throw new IOException("Failed sending due to response code: " + responseCode);
424                 this.messagesToSend.clear();
425             }
426             catch (IOException e)
427             {
428                 e.printStackTrace();
429                 //throw new TrapTransportException(message, this.state);
430                 this.delegate.ttMessagesFailedSending(this.messagesToSend, this, this.delegateContext);
431                 this.messagesToSend.clear();
432             }
433             finally
434             {
435             }
436         }
437     }
