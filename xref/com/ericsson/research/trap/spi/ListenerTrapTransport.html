38  import com.ericsson.research.trap.TrapException;
39  
40  /**
41   * An interface for a Trap Transport capable of listening for incoming connections.
42   * 
43   * @author Vladimir Katardjiev
44   */
45  public interface ListenerTrapTransport extends TrapTransport
46  {
47      
48      /**
49       * Signals to the transport that it should listen for incoming connections. The specifics of how it should listen
50       * (e.g. protocol, port, ....) are communicated in the transport configuration.
51       * 
52       * @param listener
53       *            The object that will listen to incoming connections
54       * @param context
55       *            An optional object that will be passed to the listener on each callback
56       * @throws TrapException
57       *             If an error occurred during the initialization stage of the listeners.
58       */
59      public abstract void listen(ListenerTrapTransportDelegate listener, Object context) throws TrapException;
60      
61      /**
62       * Asks this transport to generate the configuration string needed for a client to successfully connect to it. The
63       * configuration is generated and automatically inserted into the TrapConfiguration object. 
64       * 
65       * @param destination
66       *            The instance where the new configuration should be appended.
67       * @param hostname
68       *            The hostname to use for this host. This hostname will be used for any transport that has not been
69       *            explicitly configured.
70       * @throws RuntimeException
71       *             (Actual AutomaticConfigurationDisabledException) when the IP is unreachable and
72       *             hostname==null.
73       * @since 1.2.0
74       */
75      public abstract void getClientConfiguration(TrapConfiguration destination, String hostname);
76  }
