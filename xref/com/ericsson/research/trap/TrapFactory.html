51  /**
52   * TrapFactory is used as an entry point to the underlying Trap implementation. It can create listener (for servers) or
53   * client instances.
54   * 
55   * @author Vladimir Katardjiev
56   * @since 1.0
57   */
58  public class TrapFactory
59  {
60      
61      /**
62       * Creates a listener endpoint with default configuration. <b>It is up to the caller to ensure proper configuration
63       * of the listener before calling {@link TrapListener#listen(OnAccept)} </b>.
64       * 
65       * @return A newly created TrapListener instance.
66       * @throws TrapException
67       *             If an error occurred during instantiation; either if the configuration was invalid, or if the Trap
68       *             implementation was not present.
69       * @since 1.1
70       */
71      public static TrapListener createListener() throws TrapException
72      {
73          return createListener(null);
74      }
75      
76      /**
77       * Creates a listener endpoint instance, capable of accepting new incoming connections. The listener will be
78       * initialised with some default values, but will not listen for incoming connections until
79       * {@link TrapListener#listen(OnAccept)} is called.
80       * 
81       * @param configuration
82       *            The configuration string to use when creating the listener. It is strongly recommended you always have
83       *            a configuration string, even though Trap can supply default values.
84       * @return A newly created TrapListener instance.
85       * @throws TrapException
86       *             If an error occurred during instantiation; either if the configuration was invalid, or if the Trap
87       *             implementation was not present.
88       */
89      public static TrapListener createListener(String configuration) throws TrapException
90      {
91          try
92          {
93              Class<TrapListener> c = (Class<TrapListener>) Class.forName("com.ericsson.research.trap.impl.ListenerTrapEndpoint");
94              TrapListener l = c.newInstance();
95              l.configure(resolveConfiguration(configuration));
96              return l;
97          }
98          catch (Exception e)
99          {
100             throw new TrapException("Error while allocating a Listener Trap Endpoint", e);
101         }
102     }
103     
104     /**
105      * Creates a client endpoint instance, capable of creating an outgoing connection. Unlike the listener object, the
106      * client <i>must</i> be configured with proper configuration before it can connect. The listener endpoint instance
107      * is capable of creating a complete configuration profile out of the box, which can just be supplied to the client.
108      * 
109      * @param configuration
110      *            A configuration representing, at the very least, one transport address to connect to a Listener Trap
111      *            Endpoint.
112      * @param autoConfigure
113      *            Enables/disables automatic client configuration. When <i>true</i>, the client can learn of additional
114      *            transports and configuration parameters from the server. <i>false</i> makes the client ignore updated
115      *            configuration.
116      * @return A TrapEndpoint capable of opening an outgoing connection to a Listener Trap Endpoint.
117      * @throws TrapException
118      *             If an error occurred during instantiation; either if the configuration was invalid, or if the Trap
119      *             implementation was not present.
120      */
121     public static TrapClient createClient(String configuration, boolean autoConfigure) throws TrapException
122     {
123         try
124         {
125             Class<?> c = Class.forName("com.ericsson.research.trap.impl.ClientTrapEndpoint");
126             TrapClient l = (TrapClient) c.getConstructor(new Class[] { String.class, Boolean.class }).newInstance(new Object[] { resolveConfiguration(configuration), autoConfigure });
127             return l;
128         }
129         catch (Exception e)
130         {
131             throw new TrapException("Error while allocating a Client Trap Endpoint", e);
132         }
133     }
134     
135     /**
136      * Accessor for the current Trap API version. It is possible, <i>though not recommended</i> to use a different
137      * version of trap-api and trap-core. When using Maven, the versions should be matched automatically. This method
138      * has two primary purposes:
139      * <p>
140      * First, a trap-core implementation <i>should</i> verify that it matches this API version.
141      * <p>
142      * Second, someone with an unknown trap-full.jar can use this method to determine the version number.
143      * 
144      * @return The Trap API version, or "unknown" if not known. The version corresponds to the Maven version, so can be,
145      *         e.g., 1.0 or 1.1-SNAPSHOT.
146      */
147     public static String getVersion()
148     {
149         try
150         {
151             Class<?> c = Class.forName("com.ericsson.research.trap.Version");
152             Field field = c.getField("VERSION");
153             return (String) field.get(null);
154         }
155         catch (Exception e)
156         {
157             return "unknown";
158         }
159     }

161     /**
162      * Applies configuration resolution heuristics to the configuration. Scans line-by-line to detect common patterns.
163      * 
164      * @param config
165      *            The configuration string to resolve.
166      * @throws TrapException
167      */
168     protected static String resolveConfiguration(String config) throws TrapException
169     {
170         
171         if (config == null)
172             return config;
173         
174         StringBuilder rv = new StringBuilder();
175         try
176         {
177             BufferedReader br = new BufferedReader(new StringReader(config));
178             String line;
179             while ((line = br.readLine()) != null)
180             {
181                 
182                 try
183                 {
184                     
185                     if (line.startsWith("trap"))
186                     {
187                         rv.append(line);
188                         rv.append("\n");
189                         continue;
190                     }
191                     
192                     URI uri = URI.create(line);
193                     String scheme = uri.getScheme();
194                     
195                     if ("http".equals(scheme) || "https".equals(scheme))
196                     {
197                         
198                         // The common pattern of http[s]://.../trap.txt indicates we should download the config.
199                         // Query params are permitted (ignored by our code).
200                         if (uri.getPath().endsWith("trap.txt"))
201                         {
202                             
203                             try
204                             {
205                                 URL url = new URL(line);
206                                 URLConnection connection = url.openConnection();
207                                 connection.setConnectTimeout(3000);
208                                 connection.setReadTimeout(3000);
209                                 InputStream is = connection.getInputStream();
210                                 
211                                 ByteArrayOutputStream bos = new ByteArrayOutputStream();
212                                 
213                                 int c;
214                                 
215                                 while ((c = is.read()) > -1)
216                                     bos.write(c);
217                                 
218                                 String cfg = StringUtil.toUtfString(bos.toByteArray());
219                                 
220                                 // Once we've read the new config, we don't need to do anything else.
221                                 // A side-effect of this is that a single config can include multiple lines (failover-ish)
222                                 // Also, we will recurse over the config to permit resolution thereof.
223                                 return resolveConfiguration(cfg);
224                             }
225                             catch (Exception e)
226                             {
227                                 // TODO: This is a non-fatal error, but we should handle it in a more graceful manner.
228                                 e.printStackTrace();
229                                 continue;
230                             }
231                         }
232                         else
233                         {
234                             append("trap.transport.http.url=", line, rv);
235                             append("trap.transport.http.host=", uri.getHost(), rv);
236                             
237                             int port = uri.getPort();
238                             if (port == -1)
239                                 port = "http".equals(uri.getScheme()) ? 80 : 443;
240                             
241                             append("trap.transport.http.port=", Integer.toString(port), rv);
242                             
243                         }
244                     }
245                     else if ("ws".equals(scheme) || "wss".equals(scheme))
246                     {
247                         append("trap.transport.websocket.wsuri=", line, rv);
248                         append("trap.transport.websocket.host=", uri.getHost(), rv);
249                         
250                         int port = uri.getPort();
251                         if (port == -1)
252                             port = "ws".equals(uri.getScheme()) ? 80 : 443;
253                         
254                         append("trap.transport.websocket.port=", Integer.toString(port), rv);
255                         append("trap.transport.websocket.path=", uri.getPath(), rv);
256                     }
257                     else if ("socket".equals(scheme) || "sockets".equals(scheme))
258                     {
259                         append("trap.transport.socket.host=", uri.getHost(), rv);
260                         append("trap.transport.socket.port=", Integer.toString(uri.getPort()), rv);
261                         
262                         if ("sockets".equals(scheme))
263                             append("trap.transport.socket.secure=", "true", rv);
264                     }
265                 }
266                 catch (RuntimeException e)
267                 {
268                     rv.append(line);
269                     rv.append("\n");
270                 }
271                 catch (Exception e)
272                 {
273                     rv.append(line);
274                     rv.append("\n");
275                 }
276                 
277             }
278         }
279         catch (IOException e)
280         {
281             throw new TrapException("Error while parsing configuration", e);
282         }
283         
284         return rv.toString();
285     }
286     
287     private static void append(String prefix, String value, StringBuilder out)
288     {
289         if (prefix == null || value == null || out == null)
290             return;
291         out.append(prefix);
292         out.append(value);
293         out.append("\n");
294     }
