45  /**
46   * Showcases how to use Trap Channels to send multiple data flows at the same time.
47   * <p>
48   * This demo highlights how multiple priority levels can be used to designate some data as "more important", how in
49   * flight bytes can be used to distribute bandwidth (when available), and how sending is interleaved.
50   * <p>
51   * We establish a HIGH PRIORITY channel, which will be scheduled immediately for transmission. We establish a STREAM
52   * channel, which dispatches data as it is received (instead of message chunks). And we show how they interact with a
53   * normal DATA channel.
54   * <h3>Stream vs Message Performance</h3>
55   * <p>
56   * It is worth noting that the stream will have managed to echo, if one looks at the timestamp, almost 40000 bytes by
57   * the time the data channel manages to transfer 32768 bytes. This appears to be at odds with the claim that those flows
58   * are interleaved; however, one must consider that the stream is dispatched on <i>every</i> data packet, while the DATA
59   * message is dispatched once the entire transfer completes. As such, the server will not begin to echo the DATA message
60   * until the client finishes sending it. In contrast, the server can echo the STREAM data as soon as it is received.
61   * <p>
62   * If we had only done half the loop (i.e. print out the messages on the server), both the STREAM and DATA channel would
63   * reach 32768 bytes at almost the same moment. There is no inherent performance penalty on a single hop for STREAM vs
64   * MESSAGE. On a multi-hop basis there is, however. Additionally, STREAM does not need to buffer the entire message to
65   * dispatch either, which may be desirable in some circumstances.
66   * <p>
67   * {@sample ../../../src/main/java/com/ericsson/research/trap/examples/MultiplexedClient.java sample}
68   * 
69   * @author Vladimir Katardjiev
70   * @since 1.1
71   */
72  //BEGIN_INCLUDE(sample)
73  public class MultiplexedClient implements OnOpen, OnData
74  {
75      
76      public static MultiplexedClient multiplexedClient;
77      
78      // To run the demo, you must have at least one transport on the classpath.
79      public static void main(String[] args) throws Throwable
80      {
81          // Set up a server
82          MultiplexedServer.main(args);
83          
84          // Create and run our client.
85          multiplexedClient = new MultiplexedClient(MultiplexedServer.clientConfig);
86      }
87      
88      public TrapClient       client;
89      
90      // This field is only necessary for the automated tests.
91      public boolean          allDone           = false;
92      
93      public static final int HIGH_PRIO_CHANNEL = 10;
94      public static final int DATA_CHANNEL      = 1;
95      public static final int STREAM_CHANNEL    = 2;
96      
97      public MultiplexedClient(String trapCfg) throws TrapException
98      {
99          // Create a new Trap Client to the specified host.
100         this.client = TrapFactory.createClient(trapCfg, true);
101         
102         // Tell the client to use us as a delegate.
103         // We'll only handle two events this time: open and data
104         this.client.setDelegate(this, true);
105         
106         // Time to configure the channels.
107         
108         // High priority channel should get a higher priority level.
109         this.client.getChannel(HIGH_PRIO_CHANNEL).setPriority(10);
110         
111         // The two general data transfer channels will stay on the default priority (0), but we'll configure them differently.
112         // For our example, we'll use a low chunk size to force chunking on relatively small messages.
113         this.client.getChannel(DATA_CHANNEL).setChunkSize(116); // Note the chunk size includes the Trap header. So to get 100 byte data chunks, we use chunk size 116
114         this.client.getChannel(STREAM_CHANNEL).setChunkSize(116);
115         this.client.getChannel(HIGH_PRIO_CHANNEL).setChunkSize(116);
116         
117         // Limiting the in-flight bytes will also allow us to simulate somewhat network traffic.
118         // Once a channel has this number of bytes in flight, it will not dispatch any more messages to the scheduler.
119         // The scheduler will attempt to schedule the highest priority channel first, and then go down in a FCFS basis, until the transport is filled up.
120         // The scheduler resets the starting position every second, or once all channels are scheduled, whichever comes first.
121         this.client.getChannel(DATA_CHANNEL).setInFlightBytes(400);
122         this.client.getChannel(STREAM_CHANNEL).setInFlightBytes(400);
123         
124         // We'll allocate more scheduling window for the high priority channel.
125         this.client.getChannel(HIGH_PRIO_CHANNEL).setInFlightBytes(1600);
126         
127         // Finally, we'll change the STREAM channel to use Trap's streaming mode, to exemplify how that works.
128         this.client.getChannel(STREAM_CHANNEL).setStreamingMode(true);
129         
130         // This next part is mainly for simulation purposes. We need to apply a transport that provides asynchronous latency
131         // to highlight how the multiplexing works. To that end, we'll restrict all traffic to HTTP only.
132         this.client.disableAllTransports();
133         this.client.enableTransport("http");
134         
135         // Start connecting.
136         this.client.open();
137     }
138     
139     // Called when the client is open (=ready to send)
140     public void trapOpen(TrapEndpoint endpoint, Object context)
141     {
142         // We'll fire off all three of our transfers!
143         try
144         {
145             // No compression for ANY of them. We need to be inefficient for the simulation :-)
146             
147             // Send off 32768 bytes on the DATA channel. This will arrive as one trapData callback
148             // The large chunk will arrive in the middle of the stream (=interleaved).
149             this.client.send(new byte[32768], DATA_CHANNEL, false);
150             
151             // However, 65535 bytes on the STREAM channel will arrive as 656 trapData callbacks.
152             // 655 with 100 bytes each, and one with 35 bytes.
153             this.client.send(new byte[65535], STREAM_CHANNEL, false);
154             
155         }
156         catch (TrapException e)
157         {
158             e.printStackTrace();
159         }
160     }
161     
162     int     streamMessages = 0;
163     int     streamBytes    = 0;
164     boolean gotHighPrio    = false;
165     boolean gotData        = false;
166     
167     // Called when the client receives data. Since we connect to the echo server, the same message should come back.
168     public void trapData(byte[] data, int channel, TrapEndpoint endpoint, Object context)
169     {
170         switch (channel)
171         {
172             case HIGH_PRIO_CHANNEL:
173                 this.out("Got " + data.length + " bytes on the HIGH PRIORITY channel");
174                 this.gotHighPrio = true;
175                 break;
176             
177             case DATA_CHANNEL:
178                 this.out("Got " + data.length + " bytes on the DATA channel");
179                 this.gotData = true;
180                 
181                 this.out("###");
182                 this.out("We'll now send off a HIGH PRIORITY message. This message will be large and throttle the number of STREAM messages that may be sent");
183                 this.out("Note, then, that this high priority message will be echoed in significantly shorter time than the DATA channel");
184                 this.out("###");
185                 try
186                 {
187                     endpoint.send(new byte[65535], HIGH_PRIO_CHANNEL, false);
188                 }
189                 catch (TrapException e)
190                 {
191                     e.printStackTrace();
192                 }
193                 break;
194             
195             // We'll throttle the stream channel to print some log messages.
196             case STREAM_CHANNEL:
197                 this.streamMessages++;
198                 this.streamBytes += data.length;
199                 if ((this.streamMessages % 40) == 0 || this.streamBytes == 65535)
200                     this.out("I have now received " + this.streamBytes + " bytes over " + this.streamMessages + " messages on the STREAM channel");
201                 break;
202             
203             default:
204                 this.out("Got unknown message on channel " + channel);
205         }
206         
207         if (this.gotData && this.gotHighPrio && this.streamBytes == 65535)
208         {
209             this.out("All Done");
210             this.allDone = true;
211         }
212     }
213     
214     long start = System.nanoTime();
215     
216     void out(String str)
217     {
218         long curr = (System.nanoTime() - this.start) / 1000;
219         System.out.println(curr + ": " + str);
220     }
221     
222 }
