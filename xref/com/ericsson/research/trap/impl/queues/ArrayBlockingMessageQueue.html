36  import com.ericsson.research.trap.TrapEndpoint;
37  import com.ericsson.research.trap.TrapException;
38  import com.ericsson.research.trap.spi.TrapMessage;
39  import com.ericsson.research.trap.spi.queues.BlockingMessageQueue;
40  import com.ericsson.research.trap.spi.queues.MessageQueue;
41  
42  public class ArrayBlockingMessageQueue implements BlockingMessageQueue
43  {
44  	
45  	TrapMessage[]	buf;
46  	
47  	int				startIndex		= 0;
48  	int				endIndex		= 0;
49  	
50  	private long	blockingTimeout	= 30000;
51  	
52  	public ArrayBlockingMessageQueue()
53  	{
54  		this(1000);
55  	}
56  	
57  	public ArrayBlockingMessageQueue(int i)
58  	{
59  		this.buf = new TrapMessage[i];
60  	}
61  	
62  	public synchronized void put(TrapMessage m) throws TrapException
63  	{
64  		// Check if full
65  		if (this.buf[this.endIndex] != null)
66  		{
67  			long endTime = this.blockingTimeout() + System.currentTimeMillis();
68  			
69  			while (this.buf[this.endIndex] != null)
70  			{
71  				long waitTime = endTime - System.currentTimeMillis();
72  				
73  				if (waitTime > 0)
74  				{
75  					try
76  					{
77  						this.wait(waitTime);
78  					}
79  					catch (InterruptedException e)
80  					{
81  					}
82  				}
83  				else
84  				{
85  					throw new TrapException("Could not enter buffer within the time limit");
86  				}
87  			}
88  		}
89  		
90  		this.buf[this.endIndex] = m;
91  		
92  		this.endIndex = (this.endIndex + 1) % this.buf.length;
93  		
94  	}
95  	
96  	public synchronized TrapMessage peek()
97  	{
98  		return this.buf[this.startIndex];
99  	}
100 	
101 	public synchronized TrapMessage pop()
102 	{
103 		
104 		TrapMessage m = this.peek();
105 		
106 		if (m != null)
107 		{
108 			this.buf[this.startIndex] = null;
109 			this.startIndex = (this.startIndex + 1) % this.buf.length;
110 			
111 			this.notifyAll();
112 		}
113 		
114 		return m;
115 	}
116 	
117 	public synchronized int length()
118 	{
119 		int rv = 0;
120 		
121 		if (this.startIndex < this.endIndex)
122 			rv = this.endIndex - this.startIndex;
123 		else if (this.startIndex > this.endIndex)
124 			rv = (this.endIndex - this.startIndex) + this.buf.length;
125 		else if (this.buf[this.startIndex] == null)
126 			rv = 0;
127 		else
128 			rv = this.buf.length;
129 		
130 		if (rv < 0)
131 		{
132 			System.err.println("DEBUG: Mathematical impossibility. There must be a bug.");
133 		}
134 		
135 		return rv;
136 	}
137 	
138 	public long size()
139 	{
140 		return this.buf.length;
141 	}
142 	
143 	public String getQueueType()
144 	{
145 		return TrapEndpoint.BLOCKING_MESSAGE_QUEUE;
146 	}
147 	
148 	public long blockingTimeout()
149 	{
150 		return this.blockingTimeout;
151 	}
152 	
153 	public void setBlockingTimeout(long newTimeout)
154 	{
155 		this.blockingTimeout = newTimeout;
156 	}
157 	
158 	public boolean hasMoreThanOne()
159 	{
160 		return this.length() > 1;
161 	}
162 	
163 	public synchronized String toString()
164 	{
165 		
166 		StringBuffer s = new StringBuffer();
167 		s.append("AB {");
168 		s.append(this.startIndex);
169 		s.append("/");
170 		s.append(this.endIndex);
171 		s.append("} [");
172 		
173 		int numNull = 0;
174 		int nullIdx = -1;
175 		
176 		for (int i=0; i<this.buf.length; i++)
177 		{
178 			TrapMessage entry = this.buf[i];
179 			
180 			if (entry == null)
181 			{
182 				if (nullIdx < 0)
183 					nullIdx = i;
184 				
185 				numNull++;
186 				continue;
187 			}
188 			
189 			if (numNull > 0)
190 			{
191 				s.append(nullIdx);
192 				s.append("/null");
193 				if (numNull > 1)
194 				{
195 					s.append("x");
196 					s.append(numNull);
197 				}
198 				s.append(", ");
199 				numNull = 0;
200 				nullIdx = -1;
201 			}
202 			
203 			s.append(i);
204 			s.append("/");
205 			s.append(entry);
206 			s.append(", ");
207 		}
208 
209 		
210 		if (numNull > 0)
211 		{
212 			s.append(nullIdx);
213 			s.append("/null");
214 			if (numNull > 1)
215 			{
216 				s.append("x");
217 				s.append(numNull);
218 			}
219 			s.append(", ");
220 			numNull = 0;
221 			nullIdx = -1;
222 		}
223 
224 		s.deleteCharAt(s.length()-1);
225 		s.deleteCharAt(s.length()-2);
226 		s.append("]");
227 		
228 		return s.toString();
229 	}
230 
231 	public MessageQueue createNewQueue()
232 	{
233 		ArrayBlockingMessageQueue rv = new ArrayBlockingMessageQueue(this.buf.length);
234 		rv.setBlockingTimeout(this.blockingTimeout);
235 		return rv;
236 	}
237 }
