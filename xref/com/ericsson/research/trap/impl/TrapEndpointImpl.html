280         while (it.hasNext())

291         this.setKeepaliveExpiry(option);
292         
293         this.compressionEnabled = this.config.getBooleanOption(TrapEndpoint.OPTION_ENABLE_COMPRESSION, this.compressionEnabled);

318     public boolean addTransport(TrapTransport t)
319     {
320         
321         if (!t.canConnect() && !t.canListen() && t.getState() == TrapTransportState.DISCONNECTED)
322         {
323             this.logger.debug("Attempting to add transport class [{}] for handler [{}] that can neither connect nor listen. Skipping...", t.getClass().getName(), t.getTransportName());
324             return false;
325         }
326         
327         synchronized (this.transports)
328         {
329             TrapTransport old = this.transportsMap.get(t.getTransportName());
330             
331             if (old != null)
332             {
333                 
334                 int oldPrio = old.getTransportPriority();
335                 int newPrio = t.getTransportPriority();
336                 
337                 // Strict lesser equality. This allows us to be replaced by, well, the same transport.
338                 if (oldPrio < newPrio)
339                 {
340                     this.logger.debug("Attempting to add new handler for [{}] when the old handler had a higher priority. New class was [{}]/{}, old class was [{}]{}. Skipping...", new Object[] { t.getTransportName(), t.getClass().getName(), t.getTransportPriority(), old.getClass().getName(), old.getTransportPriority() });
341                     return false;
342                 }
343                 
344                 this.transports.remove(old);
345             }
346         }
347         
348         t.setFormat(this.getTrapFormat());
349         
350         this.transportsMap.put(t.getTransportName(), t);
351         synchronized (this.transports)
352         {
353             this.transports.add(t);
354         }
355         
356         return true;
357     }

368     /*
369      * (non-Javadoc)
370      * @see com.ericsson.research.trap.TrapEndpoint#close()
371      *
372      * The close implementation should send an END message to the other side.
373      * The other side will immediately send an END back and close all its transports.
374      * Consequently, this side will not be aware whether or not it will receive an END
375      * back (the transports may fail).
376      */
377     public void close()
378     {
379         
380         if (this.getState() != TrapState.OPEN)
381         {
382             // We can't close a non-open connection.
383             
384             if (this.getState() == TrapState.SLEEPING)
385             {
386                 // TODO: We should WAKE UP then DISCONNECT.
387                 this.setState(TrapState.CLOSING);
388                 this.onEnd(null, null);
389             }
390             else
391             {
392                 if ((this.getState() == TrapState.CLOSING) || (this.getState() == TrapState.CLOSED))
393                 {
394                     // Harmless call.
395                     return;
396                 }
397                 
398                 if (this.getState() == TrapState.ERROR)
399                 {
400                     // Technically harmless call, but we will log it to point out potential laziness in the coding of the error handling of our parent.
401                     this.logger.debug("Called close() on an endpoint in state ERROR. This might be caused by recovery code shared between regular and normal states");
402                     return;
403                 }
404                 
405                 if (this.getState() == TrapState.OPENING)
406                 {
407                     // TODO: This one is troublesome. close() has been called on a connection that is opening().
408                     // I think we can handle it normally (i.e. switch to closing and just end()) but it might be worth investigating
409                     // We will log.
410                     this.logger.debug("Called close() on an endpoint in state OPENING. This message is logged for debug purposes (if we don't fully close).");
411                 }
412             }
413         }
414         this.setState(TrapState.CLOSING);
415         
416         // We'll send END to the other side
417         // After that has happened, we'll close (in onend)
418         
419         try
420         {
421             this.send(this.createMessage().setOp(Operation.END));
422         }
423         catch (TrapException e)
424         {
425             this.logger.error("Setting TrapState to ERROR due to an error while disconnecting that may have left the implementation in an inconsistent state");
426             this.setState(TrapState.ERROR);
427             // TODO: Cleanup/recovery?
428         }
429         
430     }

565     Runnable sendingThread = new Runnable() {
566         public void run()
567         {
568             try
569             {
570                 for (;;)
571                 {
572                     TrapTransport first = null;
573                     
574                     if (TrapEndpointImpl.this.messageQueueRebuild)
575                     {
576                         // We don't need to synchronize this. At worst, we'll cause one more rebuild than strictly necessary
577                         // but rebuilds should be rare enough for it to not matter.
578                         TrapEndpointImpl.this.messageQueueRebuild = false;
579                         TrapEndpointImpl.this.messageQueue.rebuild(TrapEndpointImpl.this.channels);
580                     }
581 
582                     try
583                     {
584                     // Double while intentional. I let the thread linger for a moment to reduce the number of calls to the thread pool.
585                     while ((TrapEndpointImpl.this.messageQueue.peek() != null) && !TrapEndpointImpl.this.abortSending )
586                     {
587                         try
588                         {
589                             first = TrapEndpointImpl.this.availableTransports.get(0);
590                         }
591                         catch (Throwable t)
592                         {
593                             first = null;
594                         }
595                         if (first != null)
596                         {
597 
598                             if (TrapEndpointImpl.this.logger.isTraceEnabled())
599                                 TrapEndpointImpl.this.logger.trace("Now selecting transport [{}] for sending", first.getTransportName());
600 
601                             while (first.isAvailable())
602                             {
603                                 try
604                                 {
605                                     TrapMessage m = TrapEndpointImpl.this.messageQueue.peek();
606                                     if (m == null)
607                                         break;
608 
609                                     first.send(m, true);
610                                     TrapEndpointImpl.this.messageQueue.pop();
611                                 }
612                                 catch (TrapTransportException e)
613                                 {
614                                     TrapEndpointImpl.this.logger.debug(e.getMessage(), e);
615                                 }
616                                 catch (Exception e)
617                                 {
618                                     TrapEndpointImpl.this.logger.debug(e.getMessage(), e);
619 
620                                     // What should happen if we get an exception here? We don't want this loop to continue, that's for sure.
621                                     // The first transport is clearly inadequate for the task.
622                                     if (first.getState() == TrapTransportState.AVAILABLE)
623                                     {
624                                         
625                                         // Now, the problem here is that the regular API only allows us to do a graceful disconnect.
626                                         // If we do that, though, recovery code won't be initialised.
627                                         TrapEndpointImpl.this.logger.warn("Forcibly removing transport {} from available due to infinite loop protection. This code should not occur with a well-behaved transport.", first.getTransportName());
628                                         TrapEndpointImpl.this.logger.warn("Caused by {}", e.getMessage(), e);
629 
630                                         first.forceError();
631                                     }
632                                     else
633                                     {
634                                         // Transport is no longer unavailable, loop should be broken.
635                                     }
636                                 }
637                             }
638                             
639                             first.flushTransport();
640 
641                             if (!first.isAvailable())
642                             {
643                                 synchronized (TrapEndpointImpl.this)
644                                 {
645                                     if (!first.isAvailable())
646                                         TrapEndpointImpl.this.availableTransports.remove(first);
647                                 }
648                             }
649                         }
650                         else
651                             break; // Break out of the sending loop
652                     }
653                     }
654                     finally
655                     {
656                         TrapEndpointImpl.this.messageQueue.rewind();
657                     }
658 
659                     // This is an exit condition. To prevent this exit condition from creating unnecessary hits
660                     // in the thread pool, we'll let the thread yield before exiting.
661                     Thread.yield();
662                     synchronized (TrapEndpointImpl.this.sendingLock)
663                     {
664                         try
665                         {
666                             first = TrapEndpointImpl.this.availableTransports.get(0);
667                         }
668                         catch (Throwable t)
669                         {
670                         }
671 
672                         // Also block on messageQueue; this should prevent unexpected exits.
673                         synchronized (TrapEndpointImpl.this.messageQueue)
674                         {
675 
676                             if ((TrapEndpointImpl.this.messageQueue.peek() == null) || (first == null) || TrapEndpointImpl.this.abortSending)
677                             {
678                                 
679 
680                                 //System.out.println("######### Ending condition: " + (TrapEndpointImpl.this.messageQueueSize == 0) + (first == null));
681                                 TrapEndpointImpl.this.logger.trace("Send loop end: First: {}, Available: {}, EP: {}, MQ: {}", new Object[]{ first, TrapEndpointImpl.this.availableTransports, TrapEndpointImpl.this, TrapEndpointImpl.this.messageQueue });
682                                 TrapEndpointImpl.this.setSending(false);
683                                 return;
684                             }
685                         }
686                     }
687                 }
688             }
689             catch (Throwable t)
690             {
691                 TrapEndpointImpl.this.logger.error("Exception while processing sending on {}: ", TrapEndpointImpl.this, t);
692                 synchronized (TrapEndpointImpl.this.sendingLock)
693                 {
694                     TrapEndpointImpl.this.setSending(false);
695                 }
696             }
697             finally
698             {
699                 if (TrapEndpointImpl.this.abortSending)
700                 {
701                     synchronized(TrapEndpointImpl.this.sendingLock)
702                     {
703                         TrapEndpointImpl.this.abortSending = false;
704                         TrapEndpointImpl.this.sendingLock.notifyAll();
705                     }
706                 }
707                 
708                 // The final check. If we've gotten this far out, it's worth it.
709                 if ((TrapEndpointImpl.this.messageQueue.peek() != null))
710                 	kickSendingThread();
711             }
712         }
713     };

716     protected void kickSendingThread()
717     {
718         synchronized (this.sendingLock)
719         {
720             if (this.isSending())
721                 return;
722             this.setSending(true);
723         }
724         
725         ThreadPool.executeCached(this.sendingThread);
726     }

728     public synchronized void ttStateChanged(TrapTransportState newState, TrapTransportState oldState, TrapTransport transport, Object context)
729     {
730         this.logger.trace("Transport {} changed state from {} on endpoint {}", new Object[] { transport, oldState, this });
731         if (newState == TrapTransportState.AVAILABLE)
732         {
733             this.addTransportToAvailable(transport);
734             return;
735         }
736         
737         // newState is NOT available. Remove the transport from availableTransports, if it was there
738         synchronized (this)
739         {
740             this.availableTransports.remove(transport);
741         }
742         
743         // Now we'll enter failure modes.
744         if ((newState == TrapTransportState.DISCONNECTED) || (newState == TrapTransportState.ERROR))
745         {
746             
747             if ((this.getState() == TrapState.CLOSED) || (this.getState() == TrapState.CLOSING))
748             {
749                 
750                 // Make sure we update our state properly when all transports have disconnected.
751                 if (this.getState() == TrapState.CLOSING)
752                 {
753                     
754                     // Verify if this was the last open transport.
755                     for (int i = 0; i < this.transports.size(); i++)
756                     {
757                         TrapTransport t = this.transports.get(i);
758                         if ((t.getState() != TrapTransportState.ERROR) && (t.getState() != TrapTransportState.DISCONNECTED))
759                             return; // If there is at least one open transport, we won't change state.
760                     }
761                     
762                     this.setState(TrapState.CLOSED);
763                     
764                 }
765             }
766         }
767     }

1007     protected void addTransportToAvailable(TrapTransport t)
1008     {
1009         // We must perform a sorted insertion
1010         synchronized (this)
1011         {
1012             
1013             if (!t.isAvailable())
1014             {
1015                 this.logger.debug("Tried to add non-available transport to available.");
1016                 return;
1017             }
1018             
1019             boolean added = false;
1020             
1021             for (int i = 0; i < this.availableTransports.size(); i++)
1022             {
1023                 TrapTransport c = this.availableTransports.get(i);
1024                 
1025                 // Priority goes from negative to positive (most to least preferred)
1026                 // When two transports have the SAME priority, add the newest one last
1027                 
1028                 if (c.getTransportPriority() >= t.getTransportPriority())
1029                 {
1030                     this.availableTransports.add(i, t);
1031                     added = true;
1032                     break;
1033                 }
1034             }
1035             
1036             if (!added)
1037                 this.availableTransports.addLast(t);
1038             
1039             ThreadPool.executeCached(new Runnable() {
1040                 
1041                 public void run()
1042                 {
1043                     TrapTransport last = null;
1044                     
1045                     synchronized (TrapEndpointImpl.this)
1046                     {
1047                         if (TrapEndpointImpl.this.availableTransports.size() > TrapEndpointImpl.this.maxActiveTransports)
1048                         {
1049                             last = TrapEndpointImpl.this.availableTransports.removeLast();
1050                             
1051                             TrapEndpointImpl.this.logger.trace("Disconnecting [{}] as max active transports was exceeded", last);
1052                         }
1053                     }
1054                     
1055                     if (last != null)
1056                         last.disconnect();
1057                 }
1058             });
1059             
1060             if (this.getState() == TrapState.SLEEPING)
1061                 this.setState(TrapState.OPEN);
1062         }
1063         
1064         this.kickSendingThread();
1065         
1066     }

1288     protected void setState(TrapState newState)
1289     {
1290         if (newState == this.state)
1291             return; // Department of redundancy department.
1292             
1293         TrapState oldState = this.getState();
1294         this.state = newState;
1295         
1296         if (this.logger.isTraceEnabled())
1297         {
1298             this.logger.trace("{} changing state from {} to {} with transports {}", new Object[] { this, oldState, newState, this.transports });
1299         }
1300         
1301         // Unlock anyone waiting on us
1302         synchronized (this)
1303         {
1304             this.notifyAll();
1305         }
1306         
1307         // Now sort into the appropriate delegates.
1308         if (newState == TrapState.OPEN && oldState == TrapState.OPENING)
1309             this.openDelegate.trapOpen(this, this.delegateContext);
1310         else if (newState == TrapState.CLOSED && oldState == TrapState.CLOSING)
1311             this.closeDelegate.trapClose(this, this.delegateContext);
1312         else if (newState == TrapState.SLEEPING && oldState == TrapState.OPEN)
1313             this.sleepDelegate.trapSleep(this, this.delegateContext);
1314         else if (newState == TrapState.OPEN && oldState == TrapState.SLEEPING)
1315             this.wakeupDelegate.trapWakeup(this, this.delegateContext);
1316         
1317         this.statechangeDelegate.trapStateChange(newState, oldState, this, this.delegateContext);
1318         
1319         // Perform cleanup
1320         if (newState == TrapState.ERROR)
1321         {
1322             
1323             Iterator<TrapTransport> it = this.transports.iterator();
1324             
1325             while (it.hasNext())
1326                 it.next().disconnect();
1327             
1328             if (this.keepaliveTask != null)
1329                 this.keepaliveTask.cancel();
1330             
1331             this.errorDelegate.trapError(this, this.delegateContext);
1332         }
1333         
1334         // Notify messages that have not sent
1335         if ((newState == TrapState.ERROR) || (newState == TrapState.CLOSED))
1336         {
1337             LinkedList<Object> faileds = new LinkedList<Object>();
1338             
1339             while (this.messageQueue.peek() != null)
1340             {
1341                 TrapMessage msg = this.messageQueue.pop();
1342                 
1343                 if (msg == null)
1344                     break;
1345                 
1346                 if (msg instanceof TrapObjectMessage)
1347                 {
1348                     faileds.add(((TrapObjectMessage) msg).getObject());
1349                 }
1350                 else
1351                 {
1352                     faileds.add(msg.getData());
1353                 }
1354                 
1355             }
1356             
1357             if (faileds.size() > 0)
1358                 this.failedSendingDelegate.trapFailedSending(faileds, this, this.delegateContext);
1359         }
1360     }

1398         if (this.logger.isDebugEnabled())

1400

1403             this.abortSending = true;
1404             
1405             try
1406             {
1407                 
1408                 while (this.isSending())
1409                     this.sendingLock.wait(1);

1417                 }
1418                 
1419                 for (int i = 0; i < this.channels.length; i++)
1420                     if (this.channels[i] != null)
1421                         this.channels[i].rebuildMessageQueue();
1422             }
1423             catch (Exception e)
1424             {
1425 				logger.error("Unhandled exception while trying to recover failed messages; {}", e, e);
1426                 this.close();

1428             }
1429             finally
1430             {
1431                 if (this.abortSending)
1432                 {
1433                     this.abortSending = false;
1434                     this.sendingLock.notifyAll();
1435                 }
1436             }
1437         }
1438         
1439         // Restart the sending thread
1440         this.kickSendingThread();
1441         
1442     }

1465     public Callback<Boolean> isAlive(long within, boolean check, final boolean reconnect, final long timeout)
1466     {
1467         
1468         final SingleCallback<Boolean> cb = new SingleCallback<Boolean>();
1469         
1470         // Ensure lastAlive is up to date.
1471         this.lastAlive();
1472         
1473         // Define within
1474         long mustBeAliveAfter = System.currentTimeMillis() - within;
1475         
1476         // We're within the allotted time window.
1477         if (this.lastAlive > mustBeAliveAfter)
1478             return cb.callback(true);
1479         
1480         // We're not allowed to perform the liveness check...
1481         if (!check)
1482             return cb.callback(false);
1483         
1484         final Runnable recovery = new Runnable() {
1485             
1486             public void run()
1487             {
1488                 if (!reconnect)
1489                 {
1490                     cb.callback(false);
1491                     return;
1492                 }
1493                 
1494                 ThreadPool.executeCached(new Runnable() {
1495                     
1496                     public void run()
1497                     {
1498                         try
1499                         {
1500                             TrapEndpointImpl.this.setState(TrapState.SLEEPING);
1501                             TrapEndpointImpl.this.canReconnectUntil = System.currentTimeMillis() + TrapEndpointImpl.this.reconnectTimeout;
1502                             
1503                             TrapEndpointImpl.this.reconnect(timeout);
1504                             
1505                             // Reconnect will actually lock for us, so we don't need to lock.
1506                             cb.callback(TrapEndpointImpl.this.getState() == TrapState.OPEN);
1507                             return;
1508                         }
1509                         catch (Exception e)
1510                         {
1511                             TrapEndpointImpl.this.logger.error("Setting TrapEndpoint to state ERROR because reconnect failed. We don't know currently how to recover from this state, so the connection is dropped");
1512                             TrapEndpointImpl.this.setState(TrapState.ERROR);
1513                         }
1514                         
1515                         cb.callback(false);
1516                     }
1517                 });
1518             }
1519         };
1520         
1521         // TODO: The liveness check should (theoretically) be multithreaded. It also MUST be synchronized so we don't accidentally drop a transport.
1522         synchronized (this)
1523         {
1524             if (this.availableTransports.size() <= 0)
1525                 recovery.run();
1526             else
1527                 for (int i = 0; i < this.availableTransports.size(); i++)
1528                 {
1529                     this.availableTransports.get(i).isAlive(within, check, timeout).setCallback(new SingleArgumentCallback<Boolean>() {
1530                         
1531                         public void receiveSingleArgumentCallback(Boolean result)
1532                         {
1533                             if (result.booleanValue())
1534                             {
1535                                 cb.callback(true);
1536                                 return;
1537                             }
1538                             
1539                             try
1540                             {
1541                                 // Check if someone else has done a callback
1542                                 if (cb.get(0) != null)
1543                                     return;
1544                             }
1545                             catch (InterruptedException e1)
1546                             {
1547                             }
1548                             
1549                             recovery.run();
1550                             
1551                         }
1552                     });
1553                 }
1554         }
1555         
1556         return cb;
1557     }

1783 class EndpointKeepaliveTask implements Runnable
1784 {
1785     
1786     private final WeakReference<TrapEndpoint> e;
1787     
1788     public EndpointKeepaliveTask(TrapEndpoint e)
1789     {
1790         this.e = new WeakReference<TrapEndpoint>(e);
1791     }
1792     
1793     public void run()
1794     {
1795         
1796         final TrapEndpointImpl ep = (TrapEndpointImpl) this.e.get();
1797         
1798         if (ep == null)
1799             return;
1800         
1801         // Conditions that should cause this task to exit.
1802         if ((ep.getState() == TrapState.CLOSING) || (ep.getState() == TrapState.CLOSED) || (ep.getState() == TrapState.ERROR))
1803             return;
1804         
1805         if ((ep.getKeepaliveInterval() == TrapKeepalivePolicy.DISABLED) || (ep.getKeepaliveInterval() == TrapKeepalivePolicy.DEFAULT))
1806             return;
1807         
1808         // Calculate the expected time we would need for keepalives to be working
1809         long expectedTime = System.currentTimeMillis() - (ep.keepaliveInterval * 1000) - ep.keepaliveExpiry;
1810         
1811         try
1812         {
1813             // Now verify all transports are within that time.
1814             Iterator<TrapTransport> it = ep.transports.iterator();
1815             
1816             while (it.hasNext())
1817             {
1818                 final TrapTransport t = it.next();
1819                 
1820                 // Check that the transport is active
1821                 if (!t.isConnected())
1822                 {
1823                     // Inactive transports are excused from keepalives
1824                     continue;
1825                 }
1826                 
1827                 if (t.lastAlive() < (expectedTime - 10 /* Allow slight shift in expected time to account for clock drift. */))
1828                 {
1829                     // This transport is not conforming.
1830                     ep.logger.debug("Transport {} is not compliant with the keepalive timers. Last alive reported was {}, but expected {}", new Object[] { t.getTransportName(), t.lastAlive(), expectedTime });
1831                     
1832                     try
1833                     {
1834                         // Perform a manual check
1835                         t.isAlive(ep.keepaliveExpiry, true, ep.keepaliveExpiry).setCallback(new SingleArgumentCallback<Boolean>() {
1836                             
1837                             public void receiveSingleArgumentCallback(Boolean result)
1838                             {
1839                                 if (result.booleanValue())
1840                                 {
1841                                     ep.logger.info("Disconnecting transport {} because it had timed out while not performing its own checks", t.getTransportName());
1842                                     t.disconnect();
1843                                 }
1844                                 
1845                             }
1846                         });
1847                     }
1848                     catch (Exception e)
1849                     {
1850                         ep.logger.error("Exception while checking non-conforming transport", e);
1851                     }
1852                 }
1853             }
1854             
1855         }
1856         catch (ConcurrentModificationException e)
1857         {
1858             // Recurse
1859             this.run();
1860         }
1861         
1862         // Now reschedule ourselves
1863         // Performing this jump will prevent a race condition from making us spiral out of control
1864         long mTimer = ep.keepaliveInterval;
1865         
1866         if ((mTimer == TrapKeepalivePolicy.DEFAULT) || (mTimer == TrapKeepalivePolicy.DISABLED))
1867             return;
1868         
1869         ep.keepaliveTask = ThreadPool.executeAfter(this, mTimer * 1000);
1870     }
