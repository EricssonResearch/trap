41  import java.io.IOException;
42  import java.nio.channels.AsynchronousServerSocketChannel;
43  import java.nio.channels.AsynchronousSocketChannel;
44  
45  import javax.net.ssl.SSLContext;
46  
47  import com.ericsson.research.trap.nio.ServerSocket;
48  import com.ericsson.research.trap.nio.SocketFactory;
49  import com.ericsson.research.trap.nio.ServerSocket.ServerSocketHandler;
50  import com.ericsson.research.trap.nio.Socket;
51  
52  /**
53   *
54   * @author Vladimir Katardjiev
55   */
56  public class Nio2SocketFactory implements SocketFactory {
57      
58  	@Override
59      public Socket client() throws IOException {
60  
61          AsynchronousSocketChannel channel = AsynchronousSocketChannel.open();
62          return new Nio2Socket(channel);
63          
64      }
65  
66      @Override
67      public ServerSocket server(ServerSocketHandler handler) throws IOException
68      {
69          AsynchronousServerSocketChannel ss = AsynchronousServerSocketChannel.open();
70          return new Nio2ServerSocket(ss, handler);
71      }
72  
73  	@Override
74      public Socket sslClient(SSLContext sslc) throws IOException
75      {
76          AsynchronousSocketChannel channel = AsynchronousSocketChannel.open();
77  	    return new Nio2SSLSocket(channel, sslc, true);
78      }
79  
80  	@Override
81      public ServerSocket sslServer(SSLContext sslc, ServerSocketHandler handler) throws IOException
82      {
83          AsynchronousServerSocketChannel ss = AsynchronousServerSocketChannel.open();
84  	    return new Nio2SSLServerSocket(ss, handler, sslc);
85      }
86      
87  }
