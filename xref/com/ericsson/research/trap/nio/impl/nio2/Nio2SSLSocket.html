38  import java.nio.ByteBuffer;
39  import java.nio.channels.AsynchronousSocketChannel;
40  import java.nio.channels.CompletionHandler;
41  import java.util.concurrent.TimeUnit;
42  
43  import javax.net.ssl.SSLContext;
44  import javax.net.ssl.SSLEngine;
45  import javax.net.ssl.SSLEngineResult;
46  import javax.net.ssl.SSLEngineResult.HandshakeStatus;
47  import javax.net.ssl.SSLException;
48  
49  public class Nio2SSLSocket extends Nio2SocketBase
50  {
51  
52  	private static HandshakeStatus	hsStatus;
53  	private final ByteBuffer	    decodeBuf;
54  	private final SSLEngine	        engine;
55  	private final static ByteBuffer	zero	  = ByteBuffer.allocate(0);
56  
57  	public Nio2SSLSocket(AsynchronousSocketChannel sock, SSLContext sslc, boolean clientMode)
58  	{
59  		super(sock);
60  		this.engine = sslc.createSSLEngine();
61  		this.engine.setUseClientMode(clientMode);
62  		String[] procols = { "TLSv1" };
63  		this.engine.setEnabledProtocols(procols);
64  
65  		int packetBufferSize = engine.getSession().getPacketBufferSize();
66  		decodeBuf	= ByteBuffer.allocateDirect(packetBufferSize);
67  		readBuf = ByteBuffer.allocateDirect(packetBufferSize);
68  		
69  		for (int i = 0; i < 2; i++)
70  		{
71  			writeBufs[i] = ByteBuffer.allocateDirect(packetBufferSize);
72  			writeBufs[i].limit(0);
73  		}
74  		writeBufs[0].clear();
75  	}
76  
77  	@Override
78  	public synchronized void send(ByteBuffer src)
79  	{
80  
81  		if (src.remaining() == 0)
82  			return;
83  
84  		_sslSend(src);
85  	}
86  
87  	private synchronized void _sslSend(ByteBuffer src)
88  	{
89  		ByteBuffer buf = writeBufs[writeBuf];
90  
91  		if (buf.remaining() == 0)
92  			return;
93  
94  		HandshakeStatus hsStatus = null;
95  
96  		try
97  		{
98  			SSLEngineResult result = this.engine.wrap(src, buf);
99  
100 			switch (result.getStatus())
101 			{
102 			case BUFFER_OVERFLOW:
103 				return;
104 			case BUFFER_UNDERFLOW:
105 				throw new SSLException("Buffer UNDERflow on send. This should not happen.");
106 			case CLOSED:
107 				throw new SSLException("Closed SSL Engine");
108 			case OK:
109 				break;
110 			default:
111 				break;
112 
113 			}
114 			hsStatus = runDelegatedTasks(result);
115 
116 		}
117 		catch (SSLException e)
118 		{
119 			handler.error(e, this);
120 			return;
121 		}
122 
123 		needsWriting.getAndSet(true);
124 		_write();
125 
126 		switch (hsStatus)
127 		{
128 		case FINISHED:
129 			if (src == zero)
130 				handler.sent(this);
131 			break;
132 		case NOT_HANDSHAKING:
133 			send(src);
134 			break;
135 		case NEED_TASK:
136 			handler.error(new SSLException("TASK required once all were run"), this);
137 			break;
138 		case NEED_UNWRAP:
139 			break;
140 		case NEED_WRAP:
141 			_sslSend(zero);
142 			break;
143 		default:
144 			break;
145 
146 		}
147 
148 	}
149 
150 	private HandshakeStatus runDelegatedTasks(SSLEngineResult result)
151 	{
152 		HandshakeStatus hsStatus = Nio2SSLSocket.hsStatus = result.getHandshakeStatus();
153 		if (result.getHandshakeStatus() == HandshakeStatus.NEED_TASK)
154 		{
155 			Runnable runnable;
156 			while ((runnable = engine.getDelegatedTask()) != null)
157 			{
158 				runnable.run();
159 			}
160 			hsStatus = engine.getHandshakeStatus();
161 			Nio2SSLSocket.hsStatus = hsStatus;
162 		}
163 		return hsStatus;
164 	}
165 
166 	@Override
167 	public boolean isSecure()
168 	{
169 		return true;
170 	}
171 
172 	void _read()
173 	{
174 
175 		sock.read(readBuf, 128, TimeUnit.DAYS, this, new CompletionHandler<Integer, Nio2SSLSocket>()
176 		{
177 
178 			@Override
179 			public void completed(Integer result, Nio2SSLSocket attachment)
180 			{
181 				try
182 				{
183 
184 					
185 					if (result == -1)
186 					{
187 						handler.closed(Nio2SSLSocket.this);
188 						return;
189 					}
190 					
191 					readBuf.flip();
192 
193 					unwrapLoop: while (readBuf.hasRemaining())
194 					{
195 
196 						SSLEngineResult unwrap = engine.unwrap(readBuf, decodeBuf);
197 
198 						switch (unwrap.getStatus())
199 						{
200 						case BUFFER_OVERFLOW:
201 							throw new RuntimeException("Buffer overflow that should not happen");
202 
203 						case BUFFER_UNDERFLOW:
204 							readBuf.compact();
205 							break unwrapLoop;
206 
207 						case CLOSED:
208 							return;
209 						case OK:
210 							decodeBuf.flip();
211 							if (decodeBuf.hasRemaining())
212 								handler.received(decodeBuf, Nio2SSLSocket.this);
213 							decodeBuf.clear();
214 
215 							break;
216 						default:
217 							break;
218 						}
219 
220 						HandshakeStatus hsStatus = runDelegatedTasks(unwrap);
221 
222 						switch (hsStatus)
223 						{
224 						case FINISHED:
225 							synchronized (Nio2SSLSocket.this)
226 							{
227 								handler.sent(Nio2SSLSocket.this);
228 							}
229 							break;
230 						case NEED_TASK:
231 							break;
232 						case NEED_UNWRAP:
233 							break;
234 						case NEED_WRAP:
235 							_sslSend(zero);
236 							break;
237 						case NOT_HANDSHAKING:
238 							break;
239 						default:
240 							break;
241 
242 						}
243 					}
244 					if (!readBuf.hasRemaining())
245 						readBuf.clear();
246 
247 					_read();
248 				}
249 				catch (Exception exc)
250 				{
251 					handler.error(exc, Nio2SSLSocket.this);
252 					_close();
253 				}
254 			}
255 
256 			@Override
257 			public void failed(Throwable exc, Nio2SSLSocket attachment)
258 			{
259 				handler.error(exc, Nio2SSLSocket.this);
260 				_close();
261 			}
262 		});
263 	}
264 
265 	synchronized void _write()
266 	{
267 
268 		// Cut off repeat entries
269 		if (!isWriting.compareAndSet(false, true))
270 			return;
271 
272 		try
273 		{
274 
275 			int cBuf = (writeBuf + 1) % 2;
276 			ByteBuffer buf = writeBufs[cBuf];
277 
278 			if (buf.remaining() == 0)
279 			{
280 
281 				// Maybe we need to flip the buffers for more data
282 				if (!needsWriting.compareAndSet(true, false))
283 				{
284 					isWriting.getAndSet(false);
285 					return;
286 				}
287 
288 				// Clear the buffer so it can be used again
289 				buf.clear();
290 
291 				// We do. Flip the buffers to get more data to write.
292 				writeBuf = (writeBuf + 1) % 2;
293 				cBuf = (writeBuf + 1) % 2;
294 				buf = writeBufs[cBuf];
295 				buf.flip();
296 
297 				if (hsStatus == HandshakeStatus.FINISHED || hsStatus == HandshakeStatus.NOT_HANDSHAKING)
298 					handler.sent(this);
299 			}
300 
301 			final ByteBuffer mBuf = buf;
302 
303 			sock.write(mBuf, this, new CompletionHandler<Integer, Nio2SSLSocket>()
304 			{
305 
306 				@Override
307 				public void completed(Integer result, Nio2SSLSocket attachment)
308 				{
309 					isWriting.getAndSet(false);
310 					_write();
311 				}
312 
313 				@Override
314 				public void failed(Throwable exc, Nio2SSLSocket attachment)
315 				{
316 					isWriting.getAndSet(false);
317 					handler.error(exc, Nio2SSLSocket.this);
318 				}
319 			});
320 
321 		}
322 		catch (RuntimeException e)
323 		{
324 			isWriting.getAndSet(false);
325 			throw e;
326 		}
327 	}
