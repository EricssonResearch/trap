36  import java.net.InetAddress;
37  
38  /**
39   * A collection of checks for IP addresses. These utilities provide convenience
40   * methods not covered by the Java API.
41   *
42   * @author vladi
43   */
44  public abstract class IPUtil
45  {
46  
47  	static IPUtil	instance;
48  
49  	static
50  	{
51  		try
52  		{
53  			Class<?> c = Class.forName(IPUtil.class.getName() + "Impl");
54  			instance = (IPUtil) c.newInstance();
55  		}
56  		catch (Throwable t)
57  		{
58  			System.err.println("Could not initialise IPUtil Impl");
59  		}
60  	}
61  
62  	/**
63  	 * Gets an array of the publicly accesisble IP addresses of the JVM.
64  	 *
65  	 * @return an array of the publicly accesisble IP addresses of the JVM.
66  	 */
67  	public static InetAddress[] getPublicAddresses()
68  	{
69  		return instance.performGetPublicAddresses();
70  	}
71  
72  	abstract InetAddress[] performGetPublicAddresses();
73  
74  	/**
75  	 * Gets a URI-safe address of an InetAddress. For example, returns
76  	 * <code>127.0.0.1</code> for localhost in IPv4, and <code>[::1]</code> for
77  	 * localhost in IPv6.
78  	 *
79  	 * @param address
80  	 *            The address to convert
81  	 * @return A string representation of the address that is safe to insert
82  	 *         into a URI.
83  	 */
84  	public static String getAddressForURI(InetAddress address)
85  	{
86  		return instance.performGetAddressForURI(address);
87  	}
88  
89  	abstract String performGetAddressForURI(InetAddress address);
90  
91  	/**
92  	 * Checks whether an InetAddress is a local (=virtual address space). For
93  	 * example, 127.0.0.1 returns true, while 8.8.8.8 returns false.
94  	 *
95  	 * @param address
96  	 *            The address to verify.
97  	 * @return <i>true</i> if the address is defined in an address space that is
98  	 *         routable from the Internet, <i>false</i> otherwise.
99  	 */
100 	public static boolean isLocal(InetAddress address)
101 	{
102 		return instance.performIsLocal(address);
103 	}
104 
105 	abstract boolean performIsLocal(InetAddress address);
106 
107 	IPUtil()
108 	{
109 	}
110 }
