38  import java.io.IOException;
39  import java.util.HashSet;
40  
41  import com.ericsson.research.trap.spi.TrapTransport;
42  import com.ericsson.research.trap.utils.PackageScanner;
43  
44  /**
45   * Container for the automatic detection of transports. This class is primarily exposed for compatibility with platforms
46   * where automatic detection fails, e.g. Android. In such cases, applications can manually add the transport classes.
47   * 
48   * @author Vladimir Katardjiev
49   * @since 1.0
50   */
51  public final class TrapTransports
52  {
53      
54      // Prevent instantiation. This is a true static class.
55      private TrapTransports()
56      {
57      }
58      
59      private static HashSet<Class<? extends TrapTransport>> transportClasses   = new HashSet<Class<? extends TrapTransport>>();
60      private static HashSet<Class<?>>                       blacklistedClasses = new HashSet<Class<?>>();
61      private static boolean                                 useAutodiscover    = true;
62      
63      /**
64       * Generates an array of transport classes for the given classloader, plus the classes added using
65       * {@link #addTransportClass(Class)}. <i>Note that this may cause classes loaded with multiple classloaders to be
66       * present, if the classes added manually do not correspond with the given classloader</i>.
67       * 
68       * @param cl
69       *            The classloader to use to scan packages.
70       * @return An array of classes detected using automatic scanning, plus the classes added manually.
71       * @throws IOException
72       *             If scanning was not possible.
73       */
74      @SuppressWarnings("unchecked")
75      public static Class<TrapTransport>[] getTransportClasses(ClassLoader cl) throws IOException
76      {
77          HashSet<Class<? extends TrapTransport>> rv = new HashSet<Class<? extends TrapTransport>>();
78          rv.addAll(transportClasses);
79          
80          // Autodiscover, if applicable.
81          if (useAutodiscover)
82          {
83              Class<?>[] transports = PackageScanner.scan(TrapTransport.TRAP_TRANSPORT_PACKAGE, cl);
84              for (int i = 0; i < transports.length; i++)
85                  if (TrapTransport.class.isAssignableFrom(transports[i]))
86                      rv.add((Class<TrapTransport>) transports[i]);
87          }
88          
89          // Apply the blacklist
90          rv.removeAll(blacklistedClasses);
91          
92          return rv.toArray(new Class[] {});
93      }
94      
95      /**
96       * Adds a class to the available transport classes. Multiple invocations for the same class are ignored.
97       * 
98       * @param transportClass
99       *            The class to add.
100      */
101     public static void addTransportClass(Class<? extends TrapTransport> transportClass)
102     {
103         transportClasses.add(transportClass);
104         blacklistedClasses.remove(transportClass);
105     }
106     
107     /**
108      * Marks a transport class as blacklisted, i.e. it will not be returned by {@link #getTransportClasses(ClassLoader)}
109      * . This is regardless of whether or not autodiscovery would discover the transport on its own. Call
110      * {@link #addTransportClass(Class)} to negate this effect.
111      * 
112      * @param removedClass
113      *            The class to blacklist.
114      * @since 1.1
115      */
116     public static void removeTransportClass(Class<?> removedClass)
117     {
118         blacklistedClasses.add(removedClass);
119     }
120     
121     /**
122      * Sets whether to use automatically discovered transports. Set this to <i>false</i> to disable the package scanning
123      * step of Trap and only use transports explicitly configured using {@link TrapTransports#addTransportClass(Class)}.
124      * <p>
125      * The primary use case for this method is to allow the application full control over which transports are in use.
126      * Depending on the distribution of TrAP used, it may ship with one or more built-in transports. Furthermore, these
127      * built-in transports may or may not conflict with each-other and, as such, it may become difficult to determine
128      * which specific implementations are used. If full control is required, autodiscovery should be disabled.
129      * 
130      * @param useAutodiscover
131      *            <i>true</i> to enable automatic transport discovery, <i>false</i> otherwise.
132      * @since 1.1
133      */
134     public static void setUseAutodiscoveredTransports(boolean useAutodiscover)
135     {
136         TrapTransports.useAutodiscover = useAutodiscover;
137     }
138 }
