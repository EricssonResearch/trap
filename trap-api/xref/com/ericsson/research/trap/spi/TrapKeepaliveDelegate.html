38  /**
39   * Interface to be implemented by objects that want to be notified about the state of a keepalive predictor/validator.
40   * 
41   * @author Vladimir Kataradjiev
42   */
43  public interface TrapKeepaliveDelegate
44  {
45      /**
46       * Called when a {@link TrapKeepalivePredictor} notices that a keepalive it expected to receive a keepaliveSuccess
47       * call for did not occur within a certain number of milliseconds (configurable, default 5000).
48       * <p>
49       * This strongly implies the link is either dead or overloaded. The KeepalivePredictor will not, on its own, close
50       * the connection. This method is called at most one time <i>for each call to
51       * {@link TrapKeepalivePredictor#keepaliveReceived(boolean, char, int, byte[])} </i>. If keepaliveReceived() is
52       * called after receiving a notification from this method, this method may be called again.
53       * <p>
54       * Additionally, this method may be called up to once without keepaliveSuccess being called, meaning no keepalive
55       * was successful ever.
56       * <p>
57       * To summarise, what this means is:
58       * <ul>
59       * <li>This method is called every time the Predictor notices a keepalive has expired, though the exact timing is
60       * not guaranteed.
61       * <li>This method does not take into account latency of the connection/keepalive
62       * </ul>
63       * 
64       * @param predictor
65       *            The KeepalivePredictor that noticed the expiration
66       * @param msec
67       *            The number of milliseconds since the keepalive was expected. This value is the absolute time since the
68       *            keepalive was expected, and only concerns connection timeout; not load or latency.
69       */
70      public void predictedKeepaliveExpired(TrapKeepalivePredictor predictor, long msec);
71      
72      /**
73       * Callback function asking the delegate to send a keepalive message with the specified properties. The keepalive
74       * predictor itself does not directly affect the transport, but does ask for these things to happen.
75       * 
76       * @param isPing
77       *            Whether the sent message is a ping or a pong
78       * @param type
79       *            The type of keepalive message. One of '1' or '2', depending on the specific configuration.
80       * @param timer
81       *            The timer requested
82       * @param data
83       *            Keepalive predictor internal data. This should be returned unchanged.
84       */
85      public void shouldSendKeepalive(boolean isPing, char type, int timer, byte[] data);
86  }
