38  import java.util.Collection;
39  import java.util.Map;
40  
41  import com.ericsson.research.trap.TrapEndpoint;
42  import com.ericsson.research.trap.TrapException;
43  import com.ericsson.research.trap.spi.TrapMessage;
44  import com.ericsson.research.trap.spi.TrapTransport;
45  
46  /**
47   * Provides an interface by which applications can authenticate Trap sessions and messages. Trap ships with no built-in
48   * implementation of TrapAuthentication; it must be entirely user-supplied. Authentication can be set per-endpoint using
49   * {@link TrapEndpoint#setAuthentication(TrapAuthentication)}, or per-transport using
50   * {@link TrapTransport#setAuthentication(TrapAuthentication)}.
51   * <p>
52   * When set, the authentication instance will be called by the transport or endpoint whenever an unauthenticated
53   * incoming message is received, and before any potentially unauthenticated message is sent. Essentially, if the remote
54   * party can trust a message (e.g. because it's transmitted over a stateful TLS connection), this authentication will
55   * not be invoked. On the other hand, if an unsecured transport is used (e.g. plain HTTP), authentication will be
56   * invoked for every message. Most transports will lie somewhere in-between, most likely invoking authentication on
57   * session start.
58   * <p>
59   * The basic steps when Trap receives a message is to do the following:
60   * <ul>
61   * <li>Parse the incoming message, separating auth payload from data payload. Verify the message type.
62   * <li>If message, do
63   * <ul>
64   * <li>Ask the assigned TrapAuthentication instance to verify the authentication. If successful, return.
65   * <li>If unsuccessful, ask the assigned TrapAuthentication instance to
66   * {@link #createAuthenticationChallenge(TrapMessage, Map)}.
67   * <li>Send the new challenge to the remote side..
68   * </ul>
69   * <li>If challenge, do
70   * <ul>
71   * <li>Ask TrapAuthentication to {@link #createAuthenticationResponse(TrapMessage, TrapMessage, Map, Map). 
72   * <li>Send the challenge response back to the other side.
73   * </ul>
74   * </ul>
75   * 
76   * @author Vladimir Katardjiev
77   */
78  public interface TrapAuthentication
79  {
80      
81      /**
82       * Fetches a collection of keys (strings) that this TrapAuthentication instance wants from the TrapTransport. The
83       * TrapAuthentication instance must not change this collection after this call, and must not assume the transport
84       * will call this function more times than one. It can assume the transport calls this function at least once.
85       * <p>
86       * As an argument, the transport supplies the context keys available from this transport. The returned collection
87       * may not contain a key that does not exist in the <i>availableKeys</i> collection. If the TrapAuthentication's
88       * implementation requires a context value whose key not exist in <i>availableKeys</i>, it may generate a value, as
89       * long as it does not significantly compromise the integrity of the authentication.
90       * <p>
91       * If there is not enough context information for this TrapAuthentication instance to successfully work, it may
92       * throw a TrapException.
93       * 
94       * @param availableKeys
95       *            A collection containing all keys that the TrapTransport can fill in with meaningful values.
96       * @throws TrapException
97       *             If there is not enough context information for this TrapAuthentication instance to successfully work
98       * @return A collection of keys that the TrapAuthentication instance wants the TrapTransport to provide on every
99       *         call.
100      */
101     public Collection<String> getContextKeys(Collection<String> availableKeys) throws TrapException;
102     
103     /**
104      * Verifies the authentication of a message. Implementations should check that {@link TrapMessage#getAuthData()} is
105      * a valid authentication string/data. Implementations <b>must</b> treat all arguments as read-only, and may not
106      * assume any specific ordering on the received messages.
107      * 
108      * @param message
109      *            The message for which to verify the authentication for. The contents of the message MUST NOT be
110      *            modified.
111      * @param headers
112      *            A map (String, String) of eventual other transport headers. The presence and/or absence of these
113      *            headers is dependant on the transport; for example, socket will have zero headers whereas HTTP may
114      *            have plenty.. May contain any number of headers (including zero). May not be null. May not be modified
115      *            by verifyAuthentication.
116      * @param context
117      *            A non-null map of the context values requested by this TrapAuthentication in
118      *            {@link #getContextKeys(Collection)}. Every key that was returned by getContextKeys MUST be filled in.
119      * @return <i>true</i> if the authentication string is correct, <i>false</i> otherwise (incorrect, could not be
120      *         verified, etc).
121      * @throws TrapAuthenticationException
122      *             If the authentication failure is so severe that the transport should be disconnected. This exception
123      *             can be thrown to signify that the transport is behaving improperly (potential DDoS/hack) and should
124      *             be terminated.
125      */
126     public boolean verifyAuthentication(TrapMessage message, Map<String, String> headers, Map<String, Object> context) throws TrapAuthenticationException;
127     
128     /**
129      * Creates an authentication challenge. The challenge may contain any UTF-16-representable characters. It is
130      * otherwise entirely application-specified. Trap will not parse the challenge's contents.
131      * 
132      * @param src
133      *            The message that arrived with invalid authentication and needs to be challenged.
134      * @param context
135      *            A map of key/value pairs deduced from the transport and environment
136      * @return A finished authentication challenge, to be inserted into the message to the remote end.
137      */
138     public String createAuthenticationChallenge(TrapMessage src, Map<String, Object> context);
139     
140     /**
141      * Creates an authentication signature for an outgoing message. If a challenge was received, it will be included.
142      * 
143      * @param challenge
144      *            Any new challenge received; <i>null</i> if no challenge was received since the last call to this
145      *            method.
146      * @param outgoing
147      *            The outgoing message (the one to be signed). All fields should be considered immutable.
148      * @param headers
149      *            Eventual message headers
150      * @param context
151      *            A non-null map of the context values requested by this TrapAuthentication in
152      *            {@link #getContextKeys(Collection)}. Every key that was returned by getContextKeys MUST be filled in.
153      * @return An authentication response corresponding to the challenge, to be inserted into a message with no further
154      *         modifications. May be null/empty string.
155      * @throws TrapException
156      *             if no authentication response could be generated.
157      */
158     public String createAuthenticationResponse(TrapMessage challenge, TrapMessage outgoing, Map<String, String> headers, Map<String, Object> context);
159 }
