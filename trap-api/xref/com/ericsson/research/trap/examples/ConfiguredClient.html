49  /**
50   * Similar to the configured server, this sample shows how to tweak a client's configuration.
51   * <p>
52   * {@sample ../../../src/main/java/com/ericsson/research/trap/examples/ConfiguredClient.java sample}
53   * 
54   * @author Vladimir Katardjiev
55   * @since 1.1
56   */
57  //BEGIN_INCLUDE(sample)
58  public class ConfiguredClient implements OnOpen, OnData
59  {
60      
61      /** A reference to the client, to prevent GC */
62      public TrapClient client;
63      
64      /** Counter on the sent messages */
65      public int        counter = 0;
66      
67      /**
68       * Creates a new ConfiguredClient to connect to trapCfg
69       * 
70       * @param trapCfg
71       *            Trap CFG to connect to
72       * @throws TrapException
73       *             Hopefully not.
74       */
75      public ConfiguredClient(String trapCfg) throws TrapException
76      {
77          // Create a new Trap Client to the specified host.
78          // The second parameter of true tells the client to attempt to autodiscover other transports.
79          // If it was set to false, the client can only ever use the manually configured transports.
80          this.client = TrapFactory.createClient("http://trap.example.com:4000", true);
81          
82          // While the above would work if we had a server at trap.example.com, we don't have that luxury.
83          // For testing purposes, we'll use the provided trapCfg.
84          this.client = TrapFactory.createClient(trapCfg, true);
85          
86          // The same enable/disable commands as in ConfiguredServer work just as well on the client
87          this.client.disableTransport("loopback");
88          
89          // We also have the same configuration parameters in setOption/configureTransport
90          this.client.setOption(TrapEndpoint.OPTION_LOGGERPREFIX, "example");
91          
92          // Tell the client to use us as a delegate.
93          this.client.setDelegate(this, true);
94          
95          // We may need to handle a different event by another object.
96          // To that end, the second argument to setDelegate is false. This will add this delegate
97          // without removing any delegates that don't conflict with it.
98          this.client.setDelegate(new OnError() {
99              
100             public void trapError(TrapEndpoint endpoint, Object context)
101             {
102                 System.err.println("Error occurred!");
103             }
104         }, false);
105         
106         /*
107          * Keepalives are important. By default, Trap uses conservative timeouts for both clients and servers.
108          * If a client requires something else, they can reset it on-the-fly. (These methods work just as well on servers)
109          * In the example above, the worst-case scenario is that it may take up to 30 seconds to close an endpoint. 
110          * A keepalive every 15 seconds, plus 5 seconds for the server responds, allows for 20 seconds before the endpoint discovers
111          * all transports are dead and moves to SLEEPING. It is then given 10 seconds in SLEEPING to connect. Failing that, it moves to ERROR.
112          */
113         
114         // Send a keepalive roughly every 15 seconds. The exact timing may be tweaked by Trap.
115         // This value is communicated over the network (hence truncated to an int) and must be within the documented range.
116         // Note this is counted in whole seconds.
117         this.client.setKeepaliveInterval(15);
118         
119         // Allow a client up to 5000 milliseconds to receive a keepalive response. If it doesn't, it will assume the transport(s) are dead and attempt to reconnect.
120         this.client.setKeepaliveExpiry(5000);
121         
122         // Allow a client up to 15 milliseconds to reconnect at least one transport.
123         this.client.setReconnectTimeout(10000);
124         
125         // Asynchronous processing is generally faster and recommended. It also defaults to on.
126         // The following line disables it, but should be avoided.
127         this.client.setAsync(false);
128         
129         // By default, Trap allows all transports to be connected. This may be overkill for some clients as they don't saturate even a single one.
130         // Setting maxActive to one will disconnect all but the most optimum transport, as determined by the transport priority.
131         this.client.setMaxActiveTransports(1);
132         
133         // We can always reorder the priority, of course. This makes HTTP preferable to WebSocket
134         this.client.getTransport("http").setTransportPriority(TrapTransportPriority.WEBSOCKET - 1);
135         
136         // Trap also has 64 channels. Channel 0 is the system channel, though, so avoid it. With the rest, we can have some fun
137         // Channels are multiplexed. This means that we can send data on multiple channels simultaneously, i.e. transfer a large file
138         // while sending small messages. Each channel is individually in-order.
139         
140         // Channels are defined as-needed. We can send on any channel, even if we haven't configured it.
141         TrapChannel channel = this.client.getChannel(5);
142         
143         // Chunk size is the largest single Trap packet that should be sent over. Don't worry; they're reassembled on the other side!
144         // Smaller values allow more multiplexing, but increase protocol overhead. Remember, each Trap packet is 16 bytes!
145         channel.setChunkSize(1200);
146         
147         // Sets an upper limit on the number of bytes that the channel may be transmitting at any one time. Smaller values prevent this channel
148         // from crowding lower priority channels, while larger values provide more throughput especially with latency.
149         channel.setInFlightBytes(32000);
150         
151         // Trap will send from higher priority channels (up to in flight max bytes) before lower priority ones. Within the same priority group
152         // it employs round-robin queueing. Default priority is 0 (=lowest), while maximum priority of Integer.MAX_VALUE is reserved for the control channel.
153         channel.setPriority(26);
154         
155         // Channels can also be used as C-sockets. BEWARE HOWEVER! Once streaming is enabled, it MUST NOT be disabled for the channel
156         // Streaming will trigger OnData events whenever data arrives on the network, with as little buffering as possible.
157         // However, this also means Trap may (and will) chunk the data, so it may not arrive as one blob.
158         // Use with caution (e.g. video streaming)
159         channel.setStreamingMode(true);
160         
161         // Start connecting.
162         this.client.open();
163     }
164     
165     // Called when the client is open (=ready to send)
166     public void trapOpen(TrapEndpoint endpoint, Object context)
167     {
168         try
169         {
170             // Sending on a channel is easy. Just use the channel number in the send command
171             this.client.send(new byte[] {}, 5, false);
172             
173             // If you have a large message, it can be compressed! Compression is enabled/disabled on a per-message basis.
174             // By sending it on a different channel than our stream channel, we ensure it isn't chunked.
175             this.client.send(new byte[65535], 1, true);
176         }
177         catch (TrapException e)
178         {
179             // TODO Auto-generated catch block
180             e.printStackTrace();
181         }
182     }
183     
184     // Called when the client receives data. Since we connect to the echo server, the same message should come back.
185     public void trapData(byte[] data, int channel, TrapEndpoint endpoint, Object context)
186     {
187         System.out.println("Got message: " + StringUtil.toUtfString(data));
188         this.sendMessage();
189     }
190     
191     private void sendMessage()
192     {
193         String message = "Message {" + this.counter++ + "}";
194         System.out.println("Now sending: " + message);
195         
196         try
197         {
198             this.client.send(StringUtil.toUtfBytes(message));
199         }
200         catch (TrapException e)
201         {
202             e.printStackTrace();
203         }
204     }
205     
206 }
